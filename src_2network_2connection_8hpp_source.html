347  /**
348  * \brief Registers a connection type
349  * \throws CriticalException If a protocol is defined twice
350  */
351  void register_connection ( const std::string& protocol_name, const Contructor& function )
352  {
353  if ( factory.count(protocol_name) )
354  CRITICAL_ERROR("Re-registering connection protocol "+protocol_name);
355  factory[protocol_name] = function;
356  }
357 
358  /**
359  * \brief Creates a connection from its settings
360  */
361  std::unique_ptr<Connection> create(Melanobot* bot, const Settings& settings)
362  {
363  try {
364  std::string protocol = settings.get("protocol",std::string());
365  auto it = factory.find(protocol);
366  if ( it != factory.end() )
367  return it->second(bot, settings);
368  ErrorLog ("sys","Connection Error")
369  << ": Unknown connection protocol "+protocol;
370  } catch ( const CriticalException& ) {
371  throw;
372  }
373  catch ( const LocatableException& exc ) {
374  ErrorLog errlog("sys","Connection Error");
375  if ( Settings::global_settings.get("debug",0) )
376  errlog << exc.file << ':' << exc.line << ": ";
377  errlog << exc.what();
378  } catch ( const std::exception& exc ) {
379  ErrorLog ("sys","Connection Error") << exc.what();
380  }
381 
382  return nullptr;
383  }
384 
385 private:
386  ConnectionFactory(){}
387  ConnectionFactory(const ConnectionFactory&) = delete;
388  ConnectionFactory(ConnectionFactory&&) = delete;
389  ConnectionFactory& operator=(const ConnectionFactory&) = delete;
390  ConnectionFactory& operator=(ConnectionFactory&&) = delete;
391 
392  std::unordered_map<std::string,Contructor> factory;
