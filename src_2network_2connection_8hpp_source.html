429  /**
430  * \brief Registers a connection type
431  * \throws CriticalException If a protocol is defined twice
432  */
433  void register_connection ( const std::string& protocol_name, const Contructor& function )
434  {
435  if ( factory.count(protocol_name) )
436  CRITICAL_ERROR("Re-registering connection protocol "+protocol_name);
437  factory[protocol_name] = function;
438  }
439 
440  /**
441  * \brief Creates a connection from its settings
442  */
443  std::unique_ptr<Connection> create(Melanobot* bot, const Settings& settings)
444  {
445  try {
446  std::string protocol = settings.get("protocol",std::string());
447  auto it = factory.find(protocol);
448  if ( it != factory.end() )
449  return it->second(bot, settings);
450  ErrorLog ("sys","Connection Error")
451  << ": Unknown connection protocol "+protocol;
452  } catch ( const CriticalException& ) {
453  throw;
454  }
455  catch ( const LocatableException& exc ) {
456  ErrorLog errlog("sys","Connection Error");
457  if ( settings::global_settings.get("debug",0) )
458  errlog << exc.file << ':' << exc.line << ": ";
459  errlog << exc.what();
460  } catch ( const std::exception& exc ) {
461  ErrorLog ("sys","Connection Error") << exc.what();
462  }
463 
464  return nullptr;
465  }
466 
467 private:
468  ConnectionFactory(){}
469  ConnectionFactory(const ConnectionFactory&) = delete;
470  ConnectionFactory(ConnectionFactory&&) = delete;
471  ConnectionFactory& operator=(const ConnectionFactory&) = delete;
472  ConnectionFactory& operator=(ConnectionFactory&&) = delete;
473 
474  std::unordered_map<std::string,Contructor> factory;
475 };
