191 void Buffer::on_read_line(const boost::system::error_code &error)
192 {
193  if (error)
194  {
195  if ( error != boost::asio::error::eof )
196  ErrorLog("irc","Network Error") << error.message();
197  return;
198  }
199 
200  std::istream buffer_stream(&buffer_read);
201  Message msg;
202  std::getline(buffer_stream,msg.raw,'\r');
203  Log("irc",'>',1) << irc.formatter()->decode(msg.raw);
204  buffer_stream.ignore(2,'\n');
205  std::istringstream socket_stream(msg.raw);
206 
207  if ( socket_stream.peek() == ':' )
208  {
209  socket_stream.ignore();
210  socket_stream >> msg.from;
211  }
212  socket_stream >> msg.command;
213 
214  char c;
215  std::string param;
216  while ( socket_stream >> c )
217  {
218  if ( c == ':' )
219  {
220  std::getline (socket_stream, param);
221  msg.params.push_back(param);
222  break;
223  }
224  else
225  {
226  socket_stream.putback(c);
227  socket_stream >> param;
228  msg.params.push_back(param);
229  }
230  }
231 
232  msg.source = msg.destination = &irc;
233  irc.handle_message(std::move(msg));
234 
235  schedule_read();
236 }
