33 /**
34  * \brief A user visible from a connection
35  * \todo Maybe keep a pointer to the User manager and notify when \c local_id
36  * or \c channels change.
37  * \todo don't know if it might be worth avoiding direct member access
38  * things like \c name, \c host, \c global_id and \c properties can be
39  * modified without any issue, changing \c local_id might cause issues with
40  * the user manager but let's assume the Connection is able to keep them
41  * unique via protocol restrictions. Adding values to \c channels should
42  * be done using \c add_channel().
43  *
44  */
45 struct User
46 {
47  /**
48  * \brief Checks if a user matches this one
49  *
50  * In order of priority, it will look for: \c global_id, \c host, \c local_id \c name.
51  * The first of those to be non-empty on \b this, will be used to check the match
52  */
53  bool matches ( const User& user ) const
54  {
55  if ( !global_id.empty() )
56  return user.global_id == global_id;
57  if ( !host.empty() )
58  return user.host == host;
59  if ( !local_id.empty() )
60  return user.local_id == local_id;
61  return user.name == name;
62  }

107  for ( const auto& p : props )

110  if ( it != attributes.end() )

114  }
