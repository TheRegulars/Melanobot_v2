27 /**
28  * \brief Base class for handling CTCP requests
29  * \see http://www.irchelp.org/irchelp/rfc/ctcpspec.html
30  */
31 class CtcpBase : public Handler
32 {
33 public:
34 
35  CtcpBase(const std::string& ctcp, const Settings& settings, Melanobot* bot)
36  : Handler(settings,bot), ctcp(network::irc::strtoupper(ctcp))
37  {
38  if ( ctcp.empty() )
39  throw ConfigurationError();
40  }
41 
42  bool can_handle(const network::Message& msg) override
43  {
44  return authorized(msg) && !msg.params.empty() &&
45  msg.source->protocol() == "irc" &&
46  msg.channels.size() == 1 && msg.from == msg.channels[0] &&
47  network::irc::strtoupper(msg.command) == "CTCP" &&
48  network::irc::strtoupper(msg.params[0]) == ctcp;
49  }
50 
51  std::string get_property(const std::string& name) const override
52  {
53  if ( name == "ctcp" )
54  return ctcp;
55  else if ( name == "clientinfo" )
56  return clientinfo;
57  return Handler::get_property(name);
58  }
59 
60 protected:
61  /**
62  * \brief Sends a properly formatted reply corresponding to \c ctctp
63  */
64  void reply_to(const network::Message& msg,
65  const string::FormattedString& text) const override
66  {
67  string::FormattedStream s;
68  s << '\1' << ctcp << ' ' << text << '\1';
69  msg.source->command({"NOTICE",
70  {msg.from,s.str().encode(msg.source->formatter())}, priority});
71  }
72  using Handler::reply_to;
73 
74  std::string clientinfo; ///< String to be shown on clientinfo
75 
76 private:
77  std::string ctcp; ///< Name of the ctcp command to reply to
78 };

221 /**
222  * \brief Shows help about other CTCP handlers
223  * \note Strongly recommended to be enabled
224  */
225 class CtcpClientInfo: public CtcpBase
226 {
227 
228 public:
229  CtcpClientInfo ( const Settings& settings, Melanobot* bot )
230  : CtcpBase("CLIENTINFO", settings, bot)
231  {
232  clientinfo = "[command] : Shows help on CTCP commands";
233  help_group = settings.get("help_group",help_group);
234  }
235 
236 protected:
237  bool on_handle(network::Message& msg) override
238  {
239  PropertyTree props;
240  bot->populate_properties({"ctcp","clientinfo","help_group"},props);
241 
242  Properties clientinfo;
243  gather(props, clientinfo);
244 
245  if ( !clientinfo.empty() )
246  {
247  std::string query = msg.params.size() < 2 ? "" : msg.params[1];
248  auto it = clientinfo.find(network::irc::strtoupper(query));
249  if ( it != clientinfo.end() )
250  {
251  reply_to(msg,it->first+" "+it->second);
252  }
253  else
254  {
255  std::vector<std::string> ctcp;
256  ctcp.reserve(clientinfo.size());
257  for ( const auto& p : clientinfo )
258  ctcp.push_back(p.first);
259  std::sort(ctcp.begin(),ctcp.end());
260 
261  reply_to(msg,string::implode(" ",ctcp));
262  }
263  }
264 
265  return true;
266  }
267 
268 private:
269  std::string help_group; ///< Only shows help for members of this group
270 
271  /**
272  * \brief Gathers ctcp clientinfo
273  */
274  void gather(const PropertyTree& properties, Properties& out) const
275  {
276  if ( properties.get("help_group",help_group) != help_group )
277  return;
278  for ( const auto& p : properties )
279  {
280  auto name = p.second.get("ctcp","");
281  if ( !name.empty() )
282  out[name] = p.second.get("clientinfo","");
283  gather(p.second, out);
284  }
285  }
286 };
287 REGISTER_HANDLER(CtcpClientInfo,CtcpClientInfo);
