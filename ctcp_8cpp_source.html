25 /**
26  * \brief Base class for handling CTCP requests
27  * \see http://www.irchelp.org/irchelp/rfc/ctcpspec.html
28  */
29 class CtcpBase : public Handler
30 {
31 public:
32 
33  CtcpBase(const std::string& ctcp, const Settings& settings, Melanobot* bot)
34  : Handler(settings,bot), ctcp(network::irc::strtoupper(ctcp))
35  {
36  if ( ctcp.empty() )
37  throw ConfigurationError();
38  }
39 
40  bool can_handle(const network::Message& msg) override
41  {
42  return authorized(msg) && !msg.params.empty() &&
43  msg.source && msg.source->protocol() == "irc" &&
44  msg.channels.size() == 1 && msg.from == msg.channels[0] &&
45  network::irc::strtoupper(msg.command) == "CTCP" &&
46  network::irc::strtoupper(msg.params[0]) == ctcp;
47  }
48 
49 protected:
50  /**
51  * \brief Sends a properly formatted reply corresponding to \c ctctp
52  */
53  void reply_to(const network::Message& msg,
54  const string::FormattedString& text) const override
55  {
56  string::FormattedStream s;
57  s << '\1' << ctcp << ' ' << text << '\1';
58  msg.source->command({"NOTICE",
59  {msg.from,s.str().encode(msg.source->formatter())}, priority});
60  }
61  using Handler::reply_to;
62 
63 private:
64  std::string ctcp; ///< Name of the ctcp command to reply to
65 };
