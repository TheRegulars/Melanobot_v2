214 AbstractList::AbstractList(const std::string& default_trigger, bool clear,
215  const Settings& settings, Melanobot* bot)
216  : SimpleGroup(settings, bot)
217 {
218  if ( trigger.empty() )
219  trigger = name = default_trigger;
220 
221  Settings child_settings;
222  for ( const auto& p : settings )
223  {
224  if ( !p.second.data().empty() )
225  {
226  if ( p.first == "edit" )
227  child_settings.put("auth",p.second.data());
228  else if ( p.first != "trigger" && p.first != "auth" && p.first != "name" )
229  child_settings.put(p.first,p.second.data());
230  }
231  }
232 
233  children.push_back(std::make_unique<ListInsert>("+",this,child_settings,bot));
234  children.push_back(std::make_unique<ListInsert>("add",this,child_settings,bot));
235  children.push_back(std::make_unique<ListRemove>("-",this,child_settings,bot));
236  children.push_back(std::make_unique<ListRemove>("rm",this,child_settings,bot));
237  if ( clear )
238  children.push_back(std::make_unique<ListClear>(this,child_settings,bot));
239 }
