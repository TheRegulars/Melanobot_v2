32 std::string hmac_md4(const std::string& input, const std::string& key)
33 {
34  HMAC_CTX ctx;
35  HMAC_CTX_init(&ctx);
36 
37  HMAC_Init_ex(&ctx, key.data(), key.size(), EVP_md4(), nullptr);
38 
39  HMAC_Update(&ctx, reinterpret_cast<const unsigned char*>(input.data()), input.size());
40 
41  unsigned char out[16];
42  unsigned int out_size = 0;
43  HMAC_Final(&ctx, out, &out_size);
44 
45  HMAC_CTX_cleanup(&ctx);
46 
47  return std::string(out,out+out_size);
48 }

253 void XonoticConnection::command ( network::Command cmd )
254 {
255  if ( cmd.command == "rcon" )
256  {
257  if ( cmd.parameters.empty() )
258  return;
259 
260  if ( cmd.parameters[0] == "set" || cmd.parameters[0] == "alias" )
261  {
262  if ( cmd.parameters.size() != 3 )
263  {
264  ErrorLog("xon") << "Wrong parameters for \""+cmd.parameters[0]+"\"";
265  return;
266  }
267  /// \todo validate cmd.parameters[1] as a proper cvar name
268  cmd.parameters[2] = quote_string(cmd.parameters[2]);
269  }
270 
271  std::string rcon_command = string::implode(" ",cmd.parameters);
272  if ( rcon_command.empty() )
273  {
274  ErrorLog("xon") << "Empty rcon command";
275  return;
276  }
277 
278  if ( rcon_secure <= 0 )
279  {
280  Log("xon",'<',2) << formatter()->decode(rcon_command);
281  write("rcon "+rcon_password+' '+rcon_command);
282  }
283  else if ( rcon_secure == 1 )
284  {
285  Log("xon",'<',2) << formatter()->decode(rcon_command);
286  std::string argbuf = (boost::format("%ld.%06d %s")
287  % std::time(nullptr) % math::random(1000000)
288  % rcon_command).str();
289  std::string key = hmac_md4(argbuf,rcon_password);
290  write("srcon HMAC-MD4 TIME "+key+' '+argbuf);
291  }
292  else
293  {
294  Lock lock(mutex);
295  rcon_buffer.push_back(rcon_command);
296  lock.unlock();
297  request_challenge();
298  }
299  }
300  else
301  {
302  Log("xon",'<',1) << cmd.command;
303  write(cmd.command);
304  }
305 }

329 void XonoticConnection::read(const std::string& datagram)
330 {
331  if ( datagram.size() < 5 || !string::starts_with(datagram,"\xff\xff\xff\xff") )
332  {
333  ErrorLog("xon","Network Error") << "Invalid datagram: " << datagram;
334  return;
335  }
336 
337  if ( datagram[4] != 'n' )
338  {
339  network::Message msg;
340  std::istringstream socket_stream(datagram.substr(4));
341  socket_stream >> msg.command; // info, status et all
342  msg.raw = datagram;
343  if ( socket_stream.peek() == ' ' )
344  socket_stream.ignore(1);
345  msg.params.emplace_back();
346  std::getline(socket_stream,msg.params.back());
347  Log("xon",'>',4) << formatter_->decode(msg.raw);
348  handle_message(msg);
349  return;
350  }
351 
352  Lock lock(mutex);
353  std::istringstream socket_stream(line_buffer+datagram.substr(5));
354  line_buffer.clear();
355  lock.unlock();
356 
357  // convert the datagram into lines
358  std::string line;
359  while (socket_stream)
360  {
361  std::getline(socket_stream,line);
362  if (socket_stream.eof())
363  {
364  lock.lock();
365  line_buffer = line;
366  lock.unlock();
367  break;
368  }
369  network::Message msg;
370  msg.command = "n"; // log
371  msg.raw = line;
372  Log("xon",'>',4) << formatter_->decode(msg.raw); // 4 'cause spams
373  handle_message(msg);
374 
375  }
376 }

378 void XonoticConnection::handle_message(network::Message& msg)
379 {
380  if ( msg.raw.empty() )
381  return;
382 
383  if ( msg.command == "n" )
384  {
385  if ( msg.raw[0] == '\1' )
386  {
387  static std::regex regex_chat("^\1(.*)\\^7: (.*)",
388  std::regex::ECMAScript|std::regex::optimize
389  );
390  std::smatch match;
391  if ( std::regex_match(msg.raw,match,regex_chat) )
392  {
393  msg.from = match[1];
394  msg.message = match[2];
395  }
396  }
397  }
398  else if ( msg.command == "challenge" )
399  {
400  Lock lock(mutex);
401  if ( msg.params.size() != 1 || rcon_buffer.empty() )
402  {
403  ErrorLog("xon") << "Got an odd challenge from the server";
404  return;
405  }
406 
407  auto rcon_command = rcon_buffer.front();
408  if ( !rcon_command.challenged || rcon_command.timeout < network::Clock::now() )
409  {
410  lock.unlock();
411  rcon_command.challenged = false;
412  request_challenge();
413  return;
414  }
415 
416  rcon_buffer.pop_front();
417  lock.unlock();
418 
419  Log("xon",'<',2) << formatter()->decode(rcon_command.command);
420  std::string challenge = msg.params[0].substr(0,11);
421  std::string challenge_command = challenge+' '+rcon_command.command;
422  std::string key = hmac_md4(challenge_command, rcon_password);
423  write("srcon HMAC-MD4 CHALLENGE "+key+' '+challenge_command);
424 
425  request_challenge();
426 
427  return; // don't propagate challenge messages
428  }
429 
430  msg.source = msg.destination = this;
431  bot->message(msg);
432 }
