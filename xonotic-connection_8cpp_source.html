32 std::string hmac_md4(const std::string& input, const std::string& key)
33 {
34  HMAC_CTX ctx;
35  HMAC_CTX_init(&ctx);
36 
37  HMAC_Init_ex(&ctx, key.data(), key.size(), EVP_md4(), nullptr);
38 
39  HMAC_Update(&ctx, reinterpret_cast<const unsigned char*>(input.data()), input.size());
40 
41  unsigned char out[16];
42  unsigned int out_size = 0;
43  HMAC_Final(&ctx, out, &out_size);
44 
45  HMAC_CTX_cleanup(&ctx);
46 
47  return std::string(out,out+out_size);
48 }

302 {

328 void XonoticConnection::command ( network::Command cmd )
329 {
330  // discards commands sent too early
331  if ( status_ == DISCONNECTED )
332  return;
333 
334  if ( cmd.command == "rcon" )
335  {
336  if ( cmd.parameters.empty() )
337  return;
338 
339  if ( cmd.parameters[0] == "set" || cmd.parameters[0] == "alias" )
340  {
341  if ( cmd.parameters.size() != 3 )
342  {
343  ErrorLog("xon") << "Wrong parameters for \""+cmd.parameters[0]+"\"";
344  return;
345  }
346  /// \todo validate cmd.parameters[1] as a proper cvar name
347  cmd.parameters[2] = quote_string(cmd.parameters[2]);
348  }
349 
350  std::string rcon_command = string::implode(" ",cmd.parameters);
351  if ( rcon_command.empty() )
352  {
353  ErrorLog("xon") << "Empty rcon command";
354  return;
355  }
356 
357  if ( rcon_secure <= 0 )
358  {
359  Log("xon",'<',2) << formatter()->decode(rcon_command);
360  write("rcon "+rcon_password+' '+rcon_command);
361  }
362  else if ( rcon_secure == 1 )
363  {
364  Log("xon",'<',2) << formatter()->decode(rcon_command);
365  std::string argbuf = (boost::format("%ld.%06d %s")
366  % std::time(nullptr) % math::random(1000000)
367  % rcon_command).str();
368  std::string key = hmac_md4(argbuf,rcon_password);
369  write("srcon HMAC-MD4 TIME "+key+' '+argbuf);
370  }
371  else
372  {
373  Lock lock(mutex);
374  rcon_buffer.push_back(rcon_command);
375  lock.unlock();
376  request_challenge();
377  }
378  }
379  else
380  {
381  Log("xon",'<',1) << cmd.command;
382  write(cmd.command);
383  }
384 }

408 void XonoticConnection::read(const std::string& datagram)
409 {
410  if ( datagram.empty() && status_ == DISCONNECTED )
411  return;
412 
413  if ( datagram.size() < 5 || !string::starts_with(datagram,"\xff\xff\xff\xff") )
414  {
415  ErrorLog("xon","Network Error") << "Invalid datagram: " << datagram;
416  return;
417  }
418 
419  // status: WAITING -> CONNECTING
420  if ( status_ == WAITING )
421  status_ = CONNECTING;
422 
423  if ( datagram[4] != 'n' )
424  {
425  network::Message msg;
426  std::istringstream socket_stream(datagram.substr(4));
427  socket_stream >> msg.command; // info, status et all
428  msg.raw = datagram;
429  if ( socket_stream.peek() == ' ' )
430  socket_stream.ignore(1);
431  msg.params.emplace_back();
432  std::getline(socket_stream,msg.params.back());
433  Log("xon",'>',4) << formatter_->decode(msg.raw);
434  handle_message(msg);
435  return;
436  }
437 
438  Lock lock(mutex);
439  std::istringstream socket_stream(line_buffer+datagram.substr(5));
440  line_buffer.clear();
441  lock.unlock();
442 
443  // convert the datagram into lines
444  std::string line;
445  while (socket_stream)
446  {
447  std::getline(socket_stream,line);
448  if (socket_stream.eof())
449  {
450  lock.lock();
451  line_buffer = line;
452  lock.unlock();
453  break;
454  }
455  network::Message msg;
456  msg.command = "n"; // log
457  msg.raw = line;
458  Log("xon",'>',4) << formatter_->decode(msg.raw); // 4 'cause spams
459  handle_message(msg);
460  }
461 }

463 void XonoticConnection::handle_message(network::Message& msg)
464 {
465  if ( msg.raw.empty() )
466  return;
467 
468  if ( msg.command == "n" && !msg.raw.empty() )
469  {
470  // chat
471  if ( msg.raw[0] == '\1' )
472  {
473  static std::regex regex_chat("^\1(.*)\\^7: (.*)",
474  std::regex::ECMAScript|std::regex::optimize
475  );
476  std::smatch match;
477  if ( std::regex_match(msg.raw,match,regex_chat) )
478  {
479  msg.from.name = match[1];
480  msg.message = match[2];
481  msg.type = network::Message::CHAT;
482  }
483  }
484  // cvars
485  else if ( msg.raw[0] == '"' )
486  {
487  // Note cvars can actually have any characters in their identifier
488  // and values, this regex limits to:
489  // * The identifier to contain only alphanumeric or + - _
490  // * The value anything but "
491  static std::regex regex_cvar(
492  R"cvar("([^"]+)" is "([^"]*)".*)cvar",
493  std::regex::ECMAScript|std::regex::optimize
494  );
495  std::smatch match;
496  if ( std::regex_match(msg.raw, match, regex_cvar) )
497  {
498  std::string cvar_name = match[1];
499  std::string cvar_value = match[2];
500  Lock lock(mutex);
501  cvars[cvar_name] = cvars[cvar_value];
502  lock.unlock();
503  /// \todo keep in sync when update_connection() is changed
504  if ( cvar_name == "log_dest_udp" )
505  {
506  // status: INVALID -> CONNECTING
507  // status: CONNECTING -> CONNECTED + message
508  // status: CHECKING -> CONNECTED
509  if ( cvar_value != io.local_endpoint().name() )
510  {
511  update_connection();
512  }
513  else if ( status_ == CONNECTING )
514  {
515  check_user_end();
516  status_ = CONNECTED;
517  virtual_message("CONNECTED");
518  }
519  else
520  {
521  check_user_end();
522  status_ = CONNECTED;
523  }
524  }
525  }
526  }
527  // join/part
528  else if ( msg.raw[0] == ':' )
529  {
530  static std::regex regex_join(
531  R"regex(:join:(\d+):(\d+):((?:[0-9]+(?:\.[0-9]+){3})|(?:[[:xdigit:]](?::[[:xdigit:]]){7})|bot):(.*))regex",
532  std::regex::ECMAScript|std::regex::optimize
533  );
534  static std::regex regex_part(
535  R"regex(:part:(\d+))regex",
536  std::regex::ECMAScript|std::regex::optimize
537  );
538  static std::regex regex_gamestart(
539  R"regex(:gamestart:([a-z]+)_([^:]*):[0-9.]*)regex",
540  std::regex::ECMAScript|std::regex::optimize
541  );
542  static std::regex regex_name(
543  R"regex(:name:(\d+):(.*))regex",
544  std::regex::ECMAScript|std::regex::optimize
545  );
546 
547  std::smatch match;
548  if ( std::regex_match(msg.raw,match,regex_join) )
549  {
550  user_manager.users_reference().remove_if(
551  [match](const user::User& u) {
552  return u.property("entity") == match[2]; });
553  user::User user;
554  user.local_id = match[1]; // playerid
555  user.properties["entity"] = match[2]; // entity number
556  if ( match[3] != "bot" )
557  user.host = match[3];
558  user.name = match[4];
559  Lock lock(mutex);
560  user_manager.add_user(user);
561  lock.unlock();
562  msg.from = user;
563  msg.type = network::Message::JOIN;
564  Log("xon",'!',2) << "Added user " << formatter()->decode(user.name);
565  }
566  else if ( std::regex_match(msg.raw,match,regex_part) )
567  {
568  Lock lock(mutex);
569  if ( user::User *found = user_manager.user(match[1]) )
570  {
571  Log("xon",'!',2) << "Removed user " << formatter()->decode(found->name);
572  msg.from = *found;
573  user_manager.remove_user(found->local_id);
574  msg.type = network::Message::PART;
575  }
576  }
577  else if ( std::regex_match(msg.raw,match,regex_gamestart) )
578  {
579  Lock lock(mutex);
580  clear_match();
581  //check_user_start();
582  msg.command = "gamestart";
583  msg.params = { match[1], match[2] };
584  properties["gametype"] = match[1];
585  properties["map"] = match[2];
586  }
587  else if ( std::regex_match(msg.raw,match,regex_name) )
588  {
589  Lock lock(mutex);
590  if ( user::User *found = user_manager.user(match[1]) )
591  {
592  msg.from = *found;
593  msg.message = match[2];
594  msg.type = network::Message::RENAME;
595  Log("irc",'!',3) << "Renamed user " << formatter()->decode(found->name)
596  << color::nocolor << " to " << formatter()->decode(msg.message);
597  found->name = msg.message;
598  }
599  }
600  }
601  // status reply
602  else if ( status_ == CHECKING || status_ == CONNECTING )
603  {
604  static std::regex regex_status1_players(
605  R"(players: \d+ active \((\d+) max\))",
606  std::regex::ECMAScript|std::regex::optimize
607  );
608  static std::regex regex_status1(
609  "([a-z]+):\\s+(.*)",
610  std::regex::ECMAScript|std::regex::optimize
611  );
612  static std::regex regex_status1_player(
613  //rowcol IP %pl ping time frags entity name
614  R"(\^[37](\S+)\s+(\S+)\s+(\S+)\s+(\S+)\s+(\S+)\s+#([0-9]+)\s+\^7(.*))",
615  std::regex::ECMAScript|std::regex::optimize
616  );
617  std::smatch match;
618  if ( std::regex_match(msg.raw, match, regex_status1_players) )
619  {
620  cvars["g_maxplayers"] = match[1];
621  }
622  else if ( std::regex_match(msg.raw, match, regex_status1) )
623  {
624  std::string status_name = match[1];
625  std::string status_value = match[2];
626  Lock lock(mutex);
627  properties[status_name] = status_value;
628  }
629  else if ( std::regex_match(msg.raw, match, regex_status1_player) )
630  {
631  check_user(match);
632  }
633  }
634  }
635  else if ( msg.command == "challenge" )
636  {
637  Lock lock(mutex);
638  if ( msg.params.size() != 1 || rcon_buffer.empty() )
639  {
640  ErrorLog("xon") << "Got an odd challenge from the server";
641  return;
642  }
643 
644  auto rcon_command = rcon_buffer.front();
645  if ( !rcon_command.challenged || rcon_command.timeout < network::Clock::now() )
646  {
647  lock.unlock();
648  rcon_command.challenged = false;
649  request_challenge();
650  return;
651  }
652 
653  rcon_buffer.pop_front();
654  lock.unlock();
655 
656  Log("xon",'<',2) << formatter()->decode(rcon_command.command);
657  std::string challenge = msg.params[0].substr(0,11);
658  std::string challenge_command = challenge+' '+rcon_command.command;
659  std::string key = hmac_md4(challenge_command, rcon_password);
660  write("srcon HMAC-MD4 CHALLENGE "+key+' '+challenge_command);
661 
662  request_challenge();
663 
664  return; // don't propagate challenge messages
665  }
666 
667  msg.source = msg.destination = this;
668  bot->message(msg);
669 }

671 void XonoticConnection::update_connection()
672 {
673  // status: WAITING -> CONNECTING
674  if ( status_ > DISCONNECTED )
675  {
676  status_ = WAITING;
677 
678  /// \todo the host name for log_dest_udp needs to be read from settings
679  /// (and if settings don't provide it, fallback to what local_endpoint() gives)
680  /// see also in handle_message()
681  command({"rcon",{"set","log_dest_udp",io.local_endpoint().name()},1024});
682 
683  command({"rcon",{"set", "sv_eventlog", "1"},1024});
684
