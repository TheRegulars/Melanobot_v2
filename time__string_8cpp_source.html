231 std::string format(const DateTime& date_time, const std::string& fmt)
232 {
233  std::string result;
234 
235  for ( std::string::size_type i = 0; i < fmt.size(); i++ )
236  {
237  if ( fmt[i] == '\\' && i+1 < fmt.size() )
