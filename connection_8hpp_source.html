52  /**
53  * \brief Creates object from host:port string
54  */
55  explicit Server( const std::string& server )
56  : port(0)
57  {
58  auto p = server.find(':');
59  host = server.substr(0,p);
60  if ( p != std::string::npos && p < server.size()-1 && std::isdigit(server[p+1]) )
61  port = string::to_uint(server.substr(p+1));
62  }
63 
64  std::string name() const
65  {
66  std::ostringstream ss;
67  ss << host << ':' << port;
68  return ss.str();
69  }

379  * \throws CriticalException If a protocol is defined twice
380  */
381  void register_connection ( const std::string& protocol_name, const Contructor& function )
382  {
383  if ( factory.count(protocol_name) )
384  CRITICAL_ERROR("Re-registering connection protocol "+protocol_name);
385  factory[protocol_name] = function;
386  }

388  /**
389  * \brief Creates a connection from its settings
390  */
391  Connection* create(Melanobot* bot, const Settings& settings)
392  {
393  std::string protocol = settings.get("protocol",std::string());
394  auto it = factory.find(protocol);
395  if ( it != factory.end() )
396  return it->second(bot, settings);
397 
398  Log("sys",'!',0) << color::red << "Error" << color::nocolor
399  << ": Uknown connection protocol "+protocol;
400  return nullptr;
401  }
402
