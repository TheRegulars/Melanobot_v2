54  Inflector ( std::vector<std::pair<std::string,std::string>> rules, bool whole_words )
55  {
56  auto alter_regex = [whole_words](std::string&& re){
57  return whole_words ? re : '^'+re+'$';
58  };
59 
60  this->rules.reserve(rules.size());
61  for ( auto& rule : rules )
62  {
63  this->rules.emplace_back(
64  alter_regex(std::move(rule.first)),
65  std::move(rule.second)
66  );
67  }

70  /**

72  * \param phrase Phrase to inflect
73  *
