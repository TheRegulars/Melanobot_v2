56  Inflector ( std::vector<std::pair<std::string,std::string>> rules, bool whole_words )
57  {
58  auto alter_regex = [whole_words](std::string&& re){
59  return whole_words ? re : '^'+re+'$';
60  };
61 
62  this->rules.reserve(rules.size());
63  for ( auto& rule : rules )
64  {
65  this->rules.emplace_back(
66  alter_regex(std::move(rule.first)),

69  }
70  }
