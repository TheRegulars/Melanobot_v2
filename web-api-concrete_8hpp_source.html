29  * \brief Handler searching a video on Youtube
30  */
31 class SearchVideoYoutube : public SimpleJson
32 {
33 public:
34  SearchVideoYoutube(const Settings& settings, handler::HandlerContainer* parent)
35  : SimpleJson("video",settings,parent)
36  {
37  api_key = settings.get("api_key", "");
38  order = settings.get("order", order);
39  api_url = settings.get("url", api_url);
40  reply = settings.get("reply", reply);
41  not_found_reply = settings.get("not_found", not_found_reply );
42  if ( api_key.empty() || api_url.empty() || reply.empty() )
43  throw ConfigurationError();
44  }
45 
46 protected:
47  bool on_handle(network::Message& msg) override
48  {
49  request_json(msg,network::http::get(api_url,{
50  {"part", "snippet"},
51  {"type", "video" },
52  {"maxResults","1"},
53  {"order",order},
54  {"key",api_key},
55  {"q",msg.message},
56  }));
57  return true;
58  }
59 
60  void json_success(const network::Message& msg, const Settings& parsed) override
61  {
62  int found = parsed.get("pageInfo.totalResults",0);
63  if ( !found )
64  {
65  reply_to(msg,not_found_reply);
66  return;
67  }
68  string::FormatterConfig fmt;
69  string::FormatterUtf8 f8;
70  Properties prop {
71  {"videoId",parsed.get("items.0.id.videoId","")},
72  {"title",f8.decode(parsed.get("items.0.snippet.title","")).encode(fmt)},
73  {"channelTitle",f8.decode(parsed.get("items.0.snippet.channelTitle","")).encode(fmt)},
74  {"description",f8.decode(parsed.get("items.0.snippet.description","")).encode(fmt)},
75  };
76  reply_to(msg,fmt.decode(string::replace(reply,prop,"%")));
77  }
78 
79 private:
80  /**
81  * \brief API soring order
82  *
83  * Acceptable values are:
84  * * date : reverse chronological order
85  * * rating : from highest to lowest rating
86  * * relevance : relevance to the search query
87  * * title : sorted alphabetically
88  * * viewCount : from highest to lowest number of views
89  */
90  std::string order = "relevance";
91  /**
92  * \brief API key
93  */
94  std::string api_key;
95  /**
96  * \brief API URL
97  */
98  std::string api_url = "https://www.googleapis.com/youtube/v3/search";
99  /**
100  * \brief Reply to give on found
101  */
102  std::string reply = "https://www.youtube.com/watch?v=%videoId";
103  /**
104  * \brief Fixed reply to give on not found
105  */
106  std::string not_found_reply = "http://www.youtube.com/watch?v=oHg5SJYRHA0";
107 };
