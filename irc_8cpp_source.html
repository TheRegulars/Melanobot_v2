175 void Buffer::on_read_line(const boost::system::error_code &error)
176 {
177  if (error)
178  {
179  if ( error != boost::asio::error::eof )
180  ErrorLog("irc","Network Error") << error.message();
181  return;
182  }
183 
184  std::istream buffer_stream(&buffer_read);
185  Message msg;
186  std::getline(buffer_stream,msg.raw,'\r');
187  Log("irc",'>',1) << irc.formatter()->decode(msg.raw);
188  buffer_stream.ignore(2,'\n');
189  std::istringstream socket_stream(msg.raw);
190 
191  if ( socket_stream.peek() == ':' )
192  {
193  socket_stream.ignore();
194  socket_stream >> msg.from;
195  }
196  socket_stream >> msg.command;
197 
198  char c;
199  std::string param;
200  while ( socket_stream >> c )
201  {
202  if ( c == ':' )
203  {
204  std::getline (socket_stream, param);
205  msg.params.push_back(param);
206  break;
207  }
208  else
209  {
210  socket_stream.putback(c);
211  socket_stream >> param;
212  msg.params.push_back(param);
213  }
214  }
215 
216  msg.source = &irc;
217  irc.handle_message(std::move(msg));
218 
219  schedule_read();
220 }

297 void IrcConnection::command ( const Command& c )
298 {
299  Command cmd = c;
300  std::string command = strtoupper(cmd.command);
301 
302  if ( command == "PRIVMSG" || command == "NOTICE" )
303  {
304  if ( cmd.parameters.size() != 2 )
305  {
306  ErrorLog("irc") << "Wrong parameters for PRIVMSG";
307  return;
308  }
309  std::string to = strtolower(cmd.parameters[0]);
310  {
311  LOCK(mutex);
312  if ( to == current_nick_lowecase )
313  {
314  ErrorLog("irc") << "Cannot send PRIVMSG to self";
315  return;
316  }
317  }
318  if ( cmd.parameters[1].empty() )
319  {
320  ErrorLog("irc") << "Empty PRIVMSG";
321  return;
322  }
323 
324  cmd.parameters[0] = to;
325  }
326  else if ( command == "PASS" )
327  {
328  if ( status() != Connection::WAITING )
329  {
330  ErrorLog("irc") << "PASS called at a wrong time";
331  return;
332  }
333  if ( cmd.parameters.size() != 1 )
334  {
335  ErrorLog("irc") << "Ill-formed PASS";
336  return;
337  }
338  }
339  else if ( command == "NICK" )
340  {
341  if ( cmd.parameters.size() != 1 )
342  {
343  ErrorLog("irc") << "Ill-formed NICK";
344  return;
345  }
346  /// \todo validate nick
347  {
348  LOCK(mutex);
349  attempted_nick = cmd.parameters[0];
350  if ( strtolower(attempted_nick) == current_nick_lowecase )
351  {
352  attempted_nick = "";
353  current_nick = cmd.parameters[0];
354  return;
355  }
356  }
357  }
358  else if ( command == "USER" )
359  {
360  if ( cmd.parameters.size() != 4 )
361  {
362  ErrorLog("irc") << "Ill-formed USER";
363  return;
364  }
365  }
366  else if ( command == "MODE" )
367  {
368  LOCK(mutex);
369  if ( cmd.parameters.size() == 1 )
370  {
371  cmd.parameters.resize(2);
372  cmd.parameters[1] = cmd.parameters[0];
373  cmd.parameters[0] = current_nick;
374  }
375  else if ( cmd.parameters.size() != 2 ||
376  strtolower(cmd.parameters[0]) == current_nick_lowecase )
377  {
378  ErrorLog("irc") << "Ill-formed MODE";
379  return;
380  }
381  /// \todo sanitaze the mode string
382  /// \todo allow channel mode
383  }
384  else if ( command == "JOIN" )
385  {
386  if ( cmd.parameters.size() < 1 )
387  {
388  ErrorLog("irc") << "Ill-formed JOIN";
389  return;
390  }
391  /// \todo Discard already joined channels,
392  /// keep track of too many channels,
393  /// check that channel names are ok
394  /// http://tools.ietf.org/html/rfc2812#section-3.2.1
395  }
396  else if ( command == "PART" )

522 bool IrcConnection::channel_mask(const std::vector<std::string>& channels,
523  const std::string& mask) const
524 {
525  static std::regex regex_commaspace ( "(,\\s*)|(\\s+)",
526  std::regex_constants::syntax_option_type::optimize |
527  std::regex_constants::syntax_option_type::ECMAScript );
528  std::vector<std::string> masks = string::regex_split(mask,regex_commaspace);
529  for ( const auto& m : masks )
530  {
531  bool match = false;
532  if ( m == "!" )
533  match = std::any_of(channels.begin(),channels.end(),
534  [](const std::string& ch){ return !ch.empty() && ch[0]!='#'; });
535  else
536  match = string::simple_wildcard(channels,m);
537  if ( match )
538  return true;
539  }
540  return false;

549 void IrcConnection::handle_message(Message msg)
550 {
551  if ( msg.command == "001" && msg.params.size() >= 1 )
552  {
553  // RPL_WELCOME
554  mutex.lock();
555  /// \todo read current_server
556  current_nick = msg.params[0];
557  current_nick_lowecase = strtolower(current_nick);
558  for ( const auto& s : channels_to_join )
559  command({"JOIN",{s}});
560  channels_to_join.clear();
561  mutex.unlock();
562  auth();
563  connection_status = CONNECTED;
564  }
565  else if ( msg.command == "433" && msg.params.size() >= 2 )
566  {
567  // ERR_NICKNAMEINUSE
568  mutex.lock();
569  if ( strtolower(attempted_nick) == strtolower(msg.params[1]) )
570  {
571  Log("irc",'!',4) << attempted_nick << " is taken, trying a new nick";
572  /// \todo check nick max length
573  /// \todo system to try to get the best nick possible
574  Command cmd {{"NICK"},{attempted_nick+'_'},1024};
575  mutex.unlock();
576  command(cmd);
577  }
578  else
579  mutex.unlock();
580  }
581  else if ( msg.command == "464" || msg.command == "465" || msg.command == "466" )
582  {
583  // banned from server
584  reconnect();
585  }
586  else if ( msg.command == "PING" )
587  {
588  /// \todo read timeout in settings
589  /// \todo set timer to the latest server message and call PING when too old
590  command({"PONG",msg.params,1024,std::chrono::minutes(3)});
591  }
592  else if ( msg.command == "PRIVMSG" )
593  {
594  if ( msg.params.size() != 2 || msg.params[1].empty() )
595  return; // Odd PRIVMSG format
596 
597  {
598  LOCK(mutex);
599  if ( strtolower(msg.from) == current_nick_lowecase )
600  return; // received our own message for some reason, disregard
601  }
602 
603  std::string message = msg.params[1];
604

606  msg.from = nickfrom;

608 
609  if ( strtolower(msg.params[0]) == current_nick_lowecase )
610  {
611  msg.channels = { strtolower(nickfrom) };
612  msg.direct = 1;
613  }
614  else
615  {
616  msg.channels = { msg.params[0] };
617  }
618 
619  // Handle CTCP
620  if ( msg.message[0] == '\1' )
621  {
622  static std::regex regex_ctcp("\1([^ \1]+)(?: ([^\1]+))?\1",
623  std::regex_constants::syntax_option_type::optimize |
624  std::regex_constants::syntax_option_type::ECMAScript);
625  std::smatch match;
626  std::regex_match(message,match,regex_ctcp);
627  msg.message.clear();
628  std::string ctcp = strtoupper(match[1].str());
629 
630  if ( ctcp == "ACTION" )
631  {
632  msg.action = 1;
633  }
634  else
635  {
636  msg.command = "CTCP";
637  msg.params = { ctcp };
638  if ( !match[2].str().empty() )
639  msg.params.push_back(match[2].str());
640  }
641  }
642  else
643  {
644  std::regex regex_direct(string::regex_escape(current_nick)+":\\s*(.*)");

646  if ( std::regex_match(message,match,regex_direct) )
647  {
648  msg.direct = true;
649  msg.message = match[1];
650  }
651  }
652  }
653  else if ( msg.command == "NOTICE" )
654  {
655  // http://tools.ietf.org/html/rfc2812#section-3.3.2
656  // Discard because you should never send automatic replies
657  return;
658  }
