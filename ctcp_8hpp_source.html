36  * \brief Base class for handling CTCP requests
37  * \see http://www.irchelp.org/irchelp/rfc/ctcpspec.html
38  */
39 class CtcpBase : public ::handler::Handler
40 {
41 public:
42 
43  CtcpBase(const std::string& ctcp, const Settings& settings, ::handler::HandlerContainer* parent)
44  : Handler(settings,parent), ctcp(irc::strtoupper(ctcp))
45  {
46  if ( ctcp.empty() )
47  throw ConfigurationError();
48  }
49 
50  bool can_handle(const network::Message& msg) const override
51  {
52  return !msg.params.empty() &&
53  msg.source->protocol() == "irc" && msg.source == msg.destination &&
54  msg.channels.size() == 1 && msg.from.name == msg.channels[0] &&
55  irc::strtoupper(msg.command) == "CTCP" &&
56  irc::strtoupper(msg.params[0]) == ctcp;
57  }
58 
59  std::string get_property(const std::string& name) const override
60  {
61  if ( name == "ctcp" )
62  return ctcp;
63  else if ( name == "clientinfo" )
64  return clientinfo;
65  return Handler::get_property(name);
66  }
67 
68 protected:
69  /**
70  * \brief Sends a properly formatted reply corresponding to \c ctctp
71  */
72  void reply_to(const network::Message& msg,
73  network::OutputMessage output) const override
74  {
75  string::FormattedString s;
76  s << '\1' << ctcp << ' ' << output.message << '\1';
77  msg.destination->command({"NOTICE",
78  {msg.from.name,s.encode(msg.destination->formatter())}, priority});
79  }
80  using Handler::reply_to;
81 
82  std::string clientinfo; ///< String to be shown on clientinfo
83 
84 private:
85  std::string ctcp; ///< Name of the ctcp command to reply to
86 };

219 /**
220  * \brief Shows help about other CTCP handlers
221  * \note Strongly recommended to be enabled
222  */
223 class CtcpClientInfo: public CtcpBase
224 {
225 
226 public:
227  CtcpClientInfo ( const Settings& settings, ::handler::HandlerContainer* parent )
228  : CtcpBase("CLIENTINFO", settings, parent)
229  {
230  clientinfo = "[command] : Shows help on CTCP commands";
231  help_group = settings.get("help_group",help_group);
232  }
233 
234 protected:
235  bool on_handle(network::Message& msg) override
236  {
237  PropertyTree props;
238  bot->populate_properties({"ctcp","clientinfo","help_group"},props);
239 
240  Properties clientinfo;
241  gather(props, clientinfo);
242 
243  if ( !clientinfo.empty() )
244  {
245  std::string query = msg.params.size() < 2 ? "" : msg.params[1];
246  auto it = clientinfo.find(irc::strtoupper(query));
247  if ( it != clientinfo.end() )
248  {
249  reply_to(msg,it->first+" "+it->second);
250  }
251  else
252  {
253  std::vector<std::string> ctcp;
254  ctcp.reserve(clientinfo.size());
255  for ( const auto& p : clientinfo )
256  ctcp.push_back(p.first);
257  std::sort(ctcp.begin(),ctcp.end());
258 
259  reply_to(msg,string::implode(" ",ctcp));
260  }
261  }
262 
263  return true;
264  }
265 
266 private:
267  std::string help_group; ///< Only shows help for members of this group
268 
269  /**
270  * \brief Gathers ctcp clientinfo
271  */
272  void gather(const PropertyTree& properties, Properties& out) const
273  {
274  if ( properties.get("help_group",help_group) != help_group )
275  return;
276  for ( const auto& p : properties )
277  {
278  auto name = p.second.get("ctcp","");
279  if ( !name.empty() )
280  out[name] = p.second.get("clientinfo","");
281  gather(p.second, out);
282  }
283  }
