72 FormattedString FormatterIrc::decode(const std::string& source) const
73 {
74  FormattedString str;
75 
76  Utf8Parser parser;
77 
78  FormatFlags flags;
79  parser.callback_ascii = [&str,&flags,&parser](uint8_t byte)
80  {
81  // see https://github.com/myano/jenni/wiki/IRC-String-Formatting
82  switch ( byte )
83  {
84  case '\2':
85  if ( !(flags & FormatFlags::BOLD) )
86  {
87  flags |= FormatFlags::BOLD;
88  str.append(std::make_shared<Format>(flags));
89  }
90  break;
91  case '\x1f':
92  if ( !(flags & FormatFlags::UNDERLINE) )
93  {
94  flags |= FormatFlags::UNDERLINE;
95  str.append(std::make_shared<Format>(flags));
96  }
97  break;
98  case '\xf':
99  flags = FormatFlags::NO_FORMAT;
100  str.append(std::make_shared<ClearFormatting>());
101  break;
102  case '\3':
103  {
104  std::string fg;
105  if ( std::isdigit(parser.input.peek()) )
106  {
107  fg += parser.input.get();
108  if ( std::isdigit(parser.input.peek()) )
109  fg += parser.input.get();
110 
111  if ( parser.input.peek() == ',' )
112  {
113  parser.input.ignore();
114  if ( std::isdigit(parser.input.peek()) )
115  {
116  parser.input.ignore();
117  if ( std::isdigit(parser.input.peek()) )
118  parser.input.ignore();
119  }
120  else
121  parser.input.unget();
122  }
123  }
124  str.append(std::make_shared<Color>(FormatterIrc::color_from_string(fg)));
125  break;
126  }
127  case '\x1d': case '\x16':
128  // Skip unsupported format flags
129  break;
130  default:
131  str.append(std::make_shared<Character>(byte));
132  break;
133  }
134  };
135  parser.callback_utf8 = [&str](uint32_t unicode,const std::string& utf8)
136  {
137  str.append(std::make_shared<Unicode>(utf8,unicode));
138  };
139 
140  parser.parse(source);
141 
142  return str;
143 }
