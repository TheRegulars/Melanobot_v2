127 class PonyCountDown : public handler::SimpleJson

129 public:
130  PonyCountDown(const Settings& settings, handler::HandlerContainer* parent)
131  : SimpleJson("nextpony",settings,parent)
132  {
133  api_url = settings.get("url",api_url);

136  help = "Pony countdown ("+api_url+")";
137  }
138 
139 protected:
140  bool on_handle(network::Message& msg) override
141  {
142  request_json(msg,network::http::get(api_url));
143  return true;
144  }
145 
146  void json_success(const network::Message& msg, const Settings& parsed) override
147  {
148  if ( parsed.empty() )
149  return json_failure(msg);
150 
151  Properties map;
152  map["title"] = parsed.get("name","");
153  map["season"] = string::to_string(parsed.get("season",0),2);
154  map["episode"] = string::to_string(parsed.get("episode",0),2);
155  map["duration"] = parsed.get("duration","");
156  timer::DateTime time = timer::parse_time(parsed.get("time",""));
157  timer::DateTime now;
158  auto delta = time > now ? time - now : now - time;
159  map["time_delta"] = timer::duration_string(delta);
160 
161  string::FormatterConfig fmt;
162  reply_to(msg,fmt.decode(string::replace(reply,map,"%")));
163  }
164 
165  void json_failure(const network::Message& msg) override
166  {
167  reply_to(msg,not_found);
168  }
169 
170 private:
171  std::string api_url = "http://api.ponycountdown.com/next";
172  std::string reply = "%time_delta until #-b#%title#-# (S%seasonE%episode)";
173  std::string not_found = "Next episode: not soon enough D:";
174 };

281  day--;
282 
283  Properties discord {
284  {"day_name", day_names[day%day_names.size()]},
285  {"season_day", std::to_string(day%73+1)+string::English().ordinal_suffix(day%73+1)},
286  {"season", season_names[(day/73)%season_names.size()]},
287  {"yold", std::to_string(year)}
288  };
289
