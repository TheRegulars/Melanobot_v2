41 std::string urlencode ( const std::string& text )
42 {
43  std::ostringstream ss;
44  ss << std::hex << std::uppercase;
45  for ( char c : text )
46  {
47  if ( std::isalnum(c) || c == '-' || c == '.' || c == '_' || c == '~' )
48  ss << c;

50  ss << '%' << std::setw(2) << std::setfill('0') << int(c);

55 std::string build_query ( const Parameters& params )
56 {
57  std::ostringstream ss;
58  auto last = params.end(); --last;
59  for ( auto it = params.begin(); it != params.end(); ++it )
60  {
61  ss << urlencode(it->first) << '=' << urlencode(it->second);
62  if ( it != last )
63  ss << '&';
64  }
65  return ss.str();
66 }
