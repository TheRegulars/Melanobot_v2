107 std::string::size_type similarity(const std::string& s1, const std::string& s2)
108 {
109  /**
110  * \note This can be done more accurately but it doesn't matter for the
111  * purpose of where this function is called so a simple yet inaccurate
112  * algorithm is acceptable.
113  *
114  */
115 
116  using int_ = std::string::size_type;
117  int_ result = 0;
118 
119  int_ i1 = 0; // Current position on s1
120  int_ i2 = 0; // Current position on s2
121 
122  while ( i1 < s1.size() && i2 < s2.size() )
123  {
124  // find the where the first character of a string is found in the other
125  int_ next1 = s1.find(s2[i2],i1);
126  int_ next2 = s2.find(s1[i1],i2);
127 
128  // found: the character of s2 isn't too far in s1
129  if ( next1 < next2 )
130  {
131  result += i1 == next1 ? 3 : 1;
132  i1 = next1+1; // moving forward in s1
133  i2++; // using that character in s2
134  }
135  // same as above, other way round
136  else if ( next1 > next2 )
137  {
138  result += i2 == next2 ? 3 : 1;
139  i1++; // using that character in s1
140  i2 = next2+1; // moving forward in s2
141  }
142  // here next1 == next2
143  // not found
144  else if ( next1 == std::string::npos )
145  {
146  // skip this character
147  i1++;
148  i2++;
149  }
150  // found on the same place
151  else
152  {
153  result += i1 == next1 ? 3 : 1;

163 {
164  std::string output;
165 
166  /// \todo could add a failure function to the trie data for less backtracking
167  StringTrie trie = make_trie(map);
168  trie.prepend(prefix);
169  if ( trie.empty() )
170  return subject;
171 
172 
173  for ( std::string::size_type pos = 0; pos < subject.size(); pos++ )
174  {
175  auto iter = trie.root();
176  StringTrie::iterator last_valid;
177 
178  for ( ; pos < subject.size(); pos++ )
179  {
180  char c = subject[pos];

182  {
183  iter.move_down(c);
184  if ( iter.accepts() )

187  else

192  output += last_valid.data();
193  }
194  else
