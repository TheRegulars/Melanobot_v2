89  /**

93  * \param overwrite If \b true all of the properties of \c source will be used,

96  static void merge( PropertyTree& target, const PropertyTree& source, bool overwrite)
97  {
98  for ( const auto& prop : source )
99  {
100  if ( overwrite || !target.get_child_optional(prop.first) )
101  target.put(prop.first,prop.second.data());
102  }
103  }
