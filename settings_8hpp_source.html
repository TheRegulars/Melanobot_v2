81  /**
82  * \brief Merge a node with the supplied values
83  * \param target Node to be overwritten
84  * \param source Node with the properties to be read
85  * \param overwrite If \b true all of the properties of \c source will be used,
86  * if \b false, only those not already found in the tree
87  */
88  inline void merge( Settings& target, const Settings& source, bool overwrite)
89  {
90  for ( const auto& prop : source )
91  {
92  if ( overwrite || !target.get_child_optional(prop.first) )
93  target.put(prop.first,prop.second.data());
94  }
95  }

102  {
103  func(sett);
