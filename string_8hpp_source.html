694 /**

696  */
697 template<class Container, class =
698  std::enable_if_t<
699  std::is_same<
700  FormattedString,
701  std::decay_t<typename Container::value_type>
702  >::value
703  >>
704 FormattedString implode (const FormattedString& separator, const Container& elements)
705 {
706  auto iter = std::begin(elements);
707  auto end = std::end(elements);
708  if ( iter == end )
709  return {};
710 
711  FormattedString ret;
712  while ( true )
713  {
714  ret << *iter;
