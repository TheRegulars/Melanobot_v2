133 void IrcConnection::handle_message(Message msg)
134 {
135  if ( msg.command.empty() ) return;
136 
137  if ( msg.command == "001" )
138  {
139  // RPL_WELCOME: prefix 001 target :message
140  if ( msg.params.size() < 1 ) return;
141 
142  mutex.lock();
143  current_nick = msg.params[0];
144  current_server.host = msg.from;
145  current_nick_lowecase = strtolower(current_nick);
146  // copy so we can unlock before command()

155  else if ( msg.command == "005" )
156  {
157  // RPL_ISUPPORT: prefix 005 target option[=value]... :are supported by this server
158  /**
159  * \todo Use MAXCHANNELS/CHANLIMIT NICKLEN CHANNELLEN CHANTYPES PREFIX CASEMAPPING

162  {
163  auto eq = msg.params[i].find('=');

165  std::string value = eq == std::string::npos ? "1" : msg.params[i].substr(eq+1);
166  server_features[name] = value;
167  }
168  }
169  else if ( msg.command == "353" )

187  if ( !found )

197  }
198  mutex.unlock();
199  }
200  else if ( msg.command == "433" )

202  // ERR_NICKNAMEINUSE

204 
205  mutex.lock();

207  {

212  mutex.unlock();
213  command(cmd);
214  }
215  else
216  mutex.unlock();
217  }
218  else if ( msg.command == "464" || msg.command == "465" || msg.command == "466" )
219  {
220  // banned from server
221  reconnect();
222  }
223  else if ( msg.command == "PING" )
224  {
225  /// \todo read timeout in settings

329  }
330  else if ( msg.command == "PART" )
331  {
332  if ( msg.params.size() >= 1 )
333  {
334  user::User user = parse_prefix(msg.from);
335  user.channels = string::comma_split(msg.params[0]);
336  mutex.lock();
337  user::User *found = user_manager.user(user.local_id);
338  if ( found )
339  {
340  for ( const auto& c : user.channels )
341  found->remove_channel(c);
342 
343  Log("irc",'!',3) << "User " << color::dark_cyan << found->name
344  << color::dark_red << " parted " << color::nocolor
345  << string::implode(", ",user.channels);
346 
347  if ( found->channels.empty() )
348  {
349  user_manager.remove_user(found->local_id);
350  Log("irc",'!',2) << "Removed user " << color::dark_red << user.name;
351  }
352  }
353  mutex.unlock();
354  msg.channels = user.channels;
355  }
356  }
357  else if ( msg.command == "QUIT" )
358  {
359  user::User user = parse_prefix(msg.from);
360  mutex.lock();
361  user::User *found = user_manager.user(user.local_id);
362  if ( found )
363  {
364  msg.channels = found->channels;
365  user_manager.remove_user(user.local_id);
366  Log("irc",'!',2) << "Removed user " << color::dark_red << user.name;
367 
368  if ( strtolower(preferred_nick) == strtolower(user.local_id) )
369  {
370  mutex.unlock();
371  command({"NICK",{preferred_nick}});
372  // doesn't need to lock
373  // but lock again to match the following unlock
374  mutex.lock();
375  }
376  }
377  mutex.unlock();
378  }
379  else if ( msg.command == "NICK" )
380  {
381  if ( msg.params.size() == 1 )
382  {
383  user::User user = parse_prefix(msg.from);
384  mutex.lock();
385  user::User *found = user_manager.user(user.local_id);
386  if ( found )
387  {
388  msg.channels = found->channels;
389  found->name = found->local_id = msg.params[0];

402  }

575 void IrcConnection::command ( const Command& c )
576 {
577  if ( c.command.empty() ) return;
578 
579  Command cmd = c;
580  cmd.command = strtoupper(cmd.command);
581 
582 
583  if ( connection_status <= CONNECTING && cmd.command != "PASS" &&
584  cmd.command != "NICK" && cmd.command != "USER" &&
585  cmd.command != "PONG" && cmd.command != "AUTH" &&
586  cmd.command != "MODE" && cmd.command != "RECONNECT" )
587  {
588  scheduled_commands.push_back(cmd);
589  return;
590  }
591 
592  if ( cmd.command == "PRIVMSG" || cmd.command == "NOTICE" )
593  {
594  if ( cmd.parameters.size() != 2 )
595  {
596  ErrorLog("irc") << "Wrong parameters for " << cmd.command;
597  return;
598  }
599  std::string to = strtolower(cmd.parameters[0]);
600  {
601  LOCK(mutex);
602  if ( to == current_nick_lowecase )
603  {
604  ErrorLog("irc") << "Cannot send " << cmd.command << " to self";
605  return;
606  }
607  }
608  if ( cmd.parameters[1].empty() )
609  {
610  ErrorLog("irc") << "Empty " << cmd.command;
611  return;
612  }
613 
614  cmd.parameters[0] = to;
615  }
616  else if ( cmd.command == "PASS" )
617  {
618  if ( status() != Connection::WAITING )
619  {
620  ErrorLog("irc") << "PASS called at a wrong time";
621  return;
622  }
623  if ( cmd.parameters.size() != 1 )
624  {
625  ErrorLog("irc") << "Ill-formed PASS";
626  return;
627  }
628  }
629  else if ( cmd.command == "NICK" )
630  {
631  std::string new_nick;
632 
633  if ( cmd.parameters.size() == 1 )
634  {
635  std::string::size_type nick_length = string::to_uint(server_features["NICKLEN"],10,-1);
636  nick_length = std::min(nick_length,cmd.parameters[0].size());
637  for ( unsigned i = 0; i < nick_length; i++ )
638  {
639  if ( !is_nickchar(cmd.parameters[0][i]) )
640  break;
641  new_nick += cmd.parameters[0][i];
642  }
643  };
644 
645  if ( new_nick.empty() )
646  {
647  ErrorLog("irc") << "Ill-formed NICK";
648  return;
649  }
650  cmd.parameters[0] = new_nick;
651 
652  /// \todo validate nick
653  {
654  LOCK(mutex);
655  if ( new_nick == current_nick )
656  return;
657  if ( attempted_nick.empty() )
658  preferred_nick = new_nick;
659  attempted_nick = new_nick;
660  }
661  }
662  else if ( cmd.command == "USER" )
663  {
664  if ( cmd.parameters.size() != 4 )
665  {
666  ErrorLog("irc") << "Ill-formed USER";
667  return;
668  }
669  }
670  else if ( cmd.command == "MODE" )
671  {
672  LOCK(mutex);
673  if ( cmd.parameters.size() == 1 )
674  {
675  cmd.parameters.resize(2);
676  cmd.parameters[1] = cmd.parameters[0];
677  cmd.parameters[0] = current_nick;
678  }
679  else if ( cmd.parameters.size() != 2 ||
680  strtolower(cmd.parameters[0]) != current_nick_lowecase )
681  {
682  ErrorLog("irc") << "Ill-formed MODE";
683  return;
684  }
685  /// \todo sanitaze the mode string
686  /// \todo allow channel mode
687  }
688  else if ( cmd.command == "JOIN" )
689  {
690  /**
691  * \note incoming JOIN is treated differently from how the
692  * IRC specification says: each parameter is handled like
693  * a separate channel
694  */
695 
696  if ( cmd.parameters.size() < 1 )
697  {
698  ErrorLog("irc") << "Ill-formed JOIN";
699  return;
700  }
701 
702  LOCK(mutex);
703 
704  std::vector<std::string> channels;
705 
706  user::User* self_user = user_manager.user(current_nick);
707  if ( self_user )
708  {
709  std::sort(self_user->channels.begin(), self_user->channels.end());
710  std::sort(cmd.parameters.begin(), cmd.parameters.end());
711  std::transform(cmd.parameters.begin(), cmd.parameters.end(),
712  cmd.parameters.begin(), strtolower);
713 
714  std::set_difference(self_user->channels.begin(), self_user->channels.end(),
715  cmd.parameters.begin(), cmd.parameters.end(),
716  std::inserter(channels, channels.begin()));
717  }
718  else
719  {
720  channels = cmd.parameters;
721  }
722 
723  if ( channels.empty() )
724  return;
725 
726  cmd.parameters = { string::implode(",",cmd.parameters) };
727  /// \todo keep track of too many channels,
728  /// check that channel names are ok
729  /// http://tools.ietf.org/html/rfc2812#section-3.2.1
730  }
731  else if ( cmd.command == "PART" )
732  {
733  if ( cmd.parameters.size() < 1 )
734  {
735  ErrorLog("irc") << "Ill-formed PART";
736  return;
737  }
738 
739  LOCK(mutex);
740  user::User* self_user = user_manager.user(current_nick);
741  if ( self_user )
742  {
743  auto it = std::find(self_user->channels.begin(),
744  self_user->channels.end(),
745  strtolower(cmd.parameters[0]));
746  if ( it == self_user->channels.end() )
747  return;
748  }
749  }
750  else if ( cmd.command == "RECONNECT" )
751  {
752  // Custom command
753  reconnect();
754  return;
755  }
756  /* Skipped: * = maybe later o = maybe disallow completely
757  * OPER o
758  * SERVICE o
759  * QUIT *
760  * SQUIT o
761  * MODE *
762  * TOPIC *
763  * NAMES *
764  * LIST o
765  * INVITE *
766  * KICK *
767  * MOTD o
768  * LUSERS
769  * VERSION
770  * STATS
771  * LINKS
772  * TIME
773  * CONNECT o
774  * TRACE o
775  * ADMIN o
776  * INFO

858  const std::string& mask) const

861  for ( const auto& m : masks )
862  {
863  bool match = false;
864  if ( m == "!" )

866  // find channels not starting with #
867  match = std::any_of(channels.begin(),channels.end(),

869  }

873  if ( string::simple_wildcard(strtolower(chan),m) )
874  {
875  match = true;
876  break;
877  }
878  }
879 
880  if ( match )
881  return true;
882  }
883  return false;
