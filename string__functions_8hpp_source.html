33 /**
34  * \brief Turn a container into a string
35  * \pre Container::const_iterator is a ForwardIterator
36  * Container::value_type has the stream operator
37  * \note Should work on arrays just fine
38  */
39 template<class Container>
40  std::string implode(const std::string& glue, const Container& elements)
41  {
42  auto iter = std::begin(elements);
43  auto end = std::end(elements);
44  if ( iter == end )
45  return "";
46 
47  std::ostringstream ss;
48  while ( true )
49  {
50  ss << *iter;
51  ++iter;
52  if ( iter != end )
53  ss << glue;
54  else
55  break;
56  }
57 
58  return ss.str();
59  }
