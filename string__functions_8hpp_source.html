31 /**
32  * \brief Turn a container into a string
33  * \pre Container::const_iterator is a ForwardIterator
34  * Container::value_type has the stream operator
35  * \note Should work on arrays just fine
36  */
37 template<class Container>
38  std::string implode(const std::string& glue, const Container& elements)
39  {
40  auto iter = std::begin(elements);
41  auto end = std::end(elements);
42  if ( iter == end )
43  return "";
44 
45  std::ostringstream ss;
46  while ( true )
47  {
48  ss << *iter;
49  ++iter;
50  if ( iter != end )
51  ss << glue;
52  else
53  break;
54  }
55 
56  return ss.str();
57  }
