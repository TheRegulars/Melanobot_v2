235 {
236  if ( msg.command.empty() ) return;

239  {
240  // RPL_WELCOME: prefix 001 target :message
241  if ( msg.params.size() < 1 ) return;
242 
243  Lock lock(mutex);
244  current_nick = msg.params[0];
245  current_server.host = msg.from;
246  current_nick_lowecase = strtolower(current_nick);
247  // copy so we can unlock before command()

256  else if ( msg.command == "005" )
257  {
258  // RPL_ISUPPORT: prefix 005 target option[=value]... :are supported by this server
259  /**
260  * \todo Use MAXCHANNELS/CHANLIMIT NICKLEN CHANNELLEN CHANTYPES PREFIX CASEMAPPING
261  */
262  for ( std::string::size_type i = 1; i < msg.params.size()-1; i++ )
263  {
264  auto eq = msg.params[i].find('=');
265  std::string name = msg.params[i].substr(0,eq);
266  std::string value = eq == std::string::npos ? "1" : msg.params[i].substr(eq+1);
267  server_features[name] = value;
268  }
269  }
270  else if ( msg.command == "353" )
271  {
272  // RPL_NAMREPLY: prefix 353 target channel_type channel :users...
273  if ( msg.params.size() < 4 ) return;
274 
275  std::string channel = msg.params[2];
276  msg.channels = { channel };
277 
278  std::vector<std::string> users = string::regex_split(msg.params[3],"\\s+");

281  Lock lock(mutex);
282  for ( auto& user : users )
283  {
284  if ( user[0] == '@' || user[0] == '+' )
285  {
286  /// \todo maybe it would be useful to store operator/voiced status
287  user.erase(0,1);
288  }
289  user::User *found = user_manager.user(user);
290  if ( !found )
291  {
292  user::User new_user;
293  new_user.name = new_user.local_id = user;
294  found = user_manager.add_user(new_user);
295  Log("irc",'!',2) << "Added user " << color::dark_green << user;
296  }
297  found->add_channel(strtolower(channel));
298  Log("irc",'!',3) << "User " << color::dark_cyan << user
299  << color::dark_green << " joined " << color::nocolor << channel;
300  }
301  }
302  }
303  else if ( msg.command == "433" )
304  {
305  // ERR_NICKNAMEINUSE
306  if ( msg.params.size() < 2 ) return;
307 
308  Lock lock(mutex);
309  if ( strtolower(attempted_nick) == strtolower(msg.params[1]) )
310  {
311  Log("irc",'!',4) << attempted_nick << " is taken, trying a new nick";
312  /// \todo check nick max length
313  /// \todo system to try to get the best nick possible
314  Command cmd {{"NICK"},{attempted_nick+'_'},1024};
315  lock.unlock();
316  command(cmd);
317  }
318  }
319  else if ( msg.command == "464" || msg.command == "465" || msg.command == "466" )
320  {
321  // banned from server
322  reconnect();
323  }
324  else if ( msg.command == "PING" )
325  {
326  /// \todo read PING timeout in settings

328  command({"PONG",msg.params,1024,std::chrono::minutes(3)});
329  }

435  else if ( msg.command == "PART" )
436  {
437  if ( msg.params.size() >= 1 )
438  {
439  // needs to be set here because the user might be removed
440  msg.from = parse_prefix(msg.from).local_id;
441

443  remove_from_channel(msg.from,msg.channels);
444  }
445  }
446  else if ( msg.command == "QUIT" )
447  {
448  user::User user = parse_prefix(msg.from);
449  Lock lock(mutex);
450  if ( strtolower(user.local_id) == current_nick_lowecase )
451  {
452  user_manager.clear();
453  }
454  else
455  {
456  user::User *found = user_manager.user(user.local_id);
457  if ( found )
458  {
459  msg.channels = found->channels;
460  user_manager.remove_user(user.local_id);
461  Log("irc",'!',2) << "Removed user " << color::dark_red << user.name;
462 
463  if ( strtolower(preferred_nick) == strtolower(user.local_id) )
464  {
465  lock.unlock();
466  command({"NICK",{preferred_nick}});
467  }
468  }
469  }
470  }
471  else if ( msg.command == "NICK" )
472  {

474  {

476  {
477  Lock lock(mutex);

680 void IrcConnection::command ( const Command& c )
681 {
682  if ( c.command.empty() ) return;
683 
684  Command cmd = c;
685  cmd.command = strtoupper(cmd.command);
686 
687 
688  if ( connection_status <= CONNECTING && cmd.command != "PASS" &&
689  cmd.command != "NICK" && cmd.command != "USER" &&
690  cmd.command != "PONG" && cmd.command != "MODE" )
691  {
692  scheduled_commands.push_back(cmd);
693  return;
694  }
695 
696  if ( cmd.command == "PRIVMSG" || cmd.command == "NOTICE" )
697  {
698  if ( cmd.parameters.size() != 2 )
699  {
700  ErrorLog("irc") << "Wrong parameters for " << cmd.command;
701  return;
702  }
703  std::string to = strtolower(cmd.parameters[0]);
704  {
705  Lock lock(mutex);
706  if ( to == current_nick_lowecase )
707  {
708  ErrorLog("irc") << "Cannot send " << cmd.command << " to self";
709  return;
710  }
711  }
712  if ( cmd.parameters[1].empty() )
713  {
714  ErrorLog("irc") << "Empty " << cmd.command;
715  return;
716  }
717 
718  cmd.parameters[0] = to;
719  }
720  else if ( cmd.command == "PASS" )
721  {
722  if ( status() != Connection::WAITING )
723  {
724  ErrorLog("irc") << "PASS called at a wrong time";
725  return;
726  }
727  if ( cmd.parameters.size() != 1 )
728  {
729  ErrorLog("irc") << "Ill-formed PASS";
730  return;
731  }
732  }
733  else if ( cmd.command == "NICK" )
734  {
735  std::string new_nick;
736 
737  if ( cmd.parameters.size() == 1 )
738  {
739  std::string::size_type nick_length =
740  string::to_uint(server_features["NICKLEN"],10,std::string::npos);
741  nick_length = std::min(nick_length,cmd.parameters[0].size());
742  /// \note NICK validation is very basic, the spec is more precise
743  for ( unsigned i = 0; i < nick_length; i++ )
744  {
745  if ( !is_nickchar(cmd.parameters[0][i]) )
746  break;
747  new_nick += cmd.parameters[0][i];
748  }
749  };
750 
751  if ( new_nick.empty() )
752  {
753  ErrorLog("irc") << "Ill-formed NICK";
754  return;
755  }
756  cmd.parameters[0] = new_nick;
757 
758  {
759  Lock lock(mutex);
760  if ( new_nick == current_nick )
761  return;
762  if ( attempted_nick.empty() )
763  preferred_nick = new_nick;
764  attempted_nick = new_nick;
765  }
766  }
767  else if ( cmd.command == "USER" )
768  {
769  if ( cmd.parameters.size() != 4 )
770  {
771  ErrorLog("irc") << "Ill-formed USER";
772  return;
773  }
774  }
775  else if ( cmd.command == "MODE" )
776  {
777  Lock lock(mutex);
778  if ( cmd.parameters.size() == 1 )
779  {
