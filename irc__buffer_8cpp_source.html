74 void Buffer::process()
75 {
76  Command cmd;
77 
78  do
79  {
80  buffer.pop(cmd);

84  auto max_timer = Clock::now() + flood_timer_max;

86  std::this_thread::sleep_for(max_timer-flood_timer);
87  }
88  while ( cmd.timeout < Clock::now() );

155 void Buffer::on_read_line(const boost::system::error_code &error)
156 {
157  if (error)
158  {
159  if ( error != boost::asio::error::eof )
160  ErrorLog("irc","Network Error") << error.message();
161  return;
162  }
163 
164  std::istream buffer_stream(&buffer_read);
165  Message msg;
166  std::getline(buffer_stream,msg.raw,'\r');
167  Log("irc",'>',1) << irc.formatter()->decode(msg.raw);
168  buffer_stream.ignore(2,'\n');
169  std::istringstream socket_stream(msg.raw);
170 
171  if ( socket_stream.peek() == ':' )
172  {
173  socket_stream.ignore();
174  socket_stream >> msg.from;
175  }
176  socket_stream >> msg.command;
177 
178  char c;
179  std::string param;
180  while ( socket_stream >> c )
181  {
182  if ( c == ':' )
183  {
184  std::getline (socket_stream, param);
185  msg.params.push_back(param);
186  break;
187  }
188  else
189  {
190  socket_stream.putback(c);
191  socket_stream >> param;
192  msg.params.push_back(param);
193  }
194  }
195 
196  msg.source = &irc;
197  irc.handle_message(std::move(msg));
198 
199  schedule_read();
200 }
