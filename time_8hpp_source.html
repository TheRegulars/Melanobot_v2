91  /**
92  * \brief Start the timer with the already-set duration
93  * \note If \c timeout is zero, \c repeating will be ignored
94  * \return \b true on success
95  */
96  bool start()
97  {
98  if ( running() || !action )
99  return false;
100  if ( timeout == duration_type::zero() )
101  callback(action);
102  thread = std::move(std::thread([this]{run();}));
103  return running();
104  }

107  * \brief Whether the timer has been started and is still running
108  */
109  bool running() const
110  {
111  return thread.joinable();
112  }

114  /**
115  * \brief Stop the timer and set a new timeout
116  * \return \b true on success
117  */
118  bool reset(duration_type timeout)
119  {
120  stop();

122  return start();
123  }

125  /**
126  * \brief Stop the timer
127  */
128  void stop()
129  {
130  if ( running() )
131  {
132  active = false;
133  condition.notify_all();
134  try {
135  thread.join();
136  // catch in the case the thread joined early
137  }catch(const std::invalid_argument&) {}
138  }
139  }
140 
141  private:
142  duration_type timeout; ///< Timer duration
143  std::atomic<bool> repeating{false};///< Whether it should start over once finished
144  std::atomic<bool> active{false}; ///< Whether it's currently running
145  std::condition_variable condition; ///< Wait condition
146  function_type action; ///< Executed on timeout
147  std::thread thread; ///< Thread where the callback is executed
148 
149  /**
150  * \brief Runs the timer
151  */
152  void run()
153  {
154  std::mutex mutex;
155  active = true;
156  while ( active && repeating )
157  {
158  Lock lock(mutex);
159  condition.wait_until(lock,clock_type::now()+timeout,[this]{return !active;});
160  if ( active )
161  action();
162  }
163  active = false;
164  }

587  DateTime& operator+= (const std::chrono::duration<Rep,Period>& dur) noexcept
588  {
589  if ( dur < std::chrono::duration<Rep,Period>::zero() )
590  return *this -= -dur;
591  if ( dur == std::chrono::duration<Rep,Period>::zero() )
592  return *this;
593 
594  // adding a positive duration
595 
596  auto ms = std::chrono::duration_cast<milliseconds>(dur).count();
597  decltype(ms) mask = 1;
598 
599  // add time
600  add_helper(1000,mask,ms,milliseconds_);
601  add_helper(60,mask,ms,second_);
602  add_helper(60,mask,ms,minute_);
603  add_helper(24,mask,ms,hour_);
604 
605  // now the time has been added correctly, ms is rounded to the day
606  // and the date needs to be adjusted
607  auto d = ms / mask;
608 
609  // round to the next year
610  if ( d >= year_days(year_) - year_day() )
611  {
612  d -= year_days(year_) - year_day();
613  set_date(year_+1,Month::JANUARY,days(1));
614  }
615 
616  // advance whole years
617  while ( d >= year_days(year_) )
618  {
619  d -= year_days(year_);
620  year_++;
621  }
622 
623  // advance whole months
624  while ( d >= month_days(month_) )
625  {
626  d -= month_days(month_);
627  month_++;
628  }
629 
630  // advance whole days
631  day_ += d;
632 
633  // handle day overflow
634  if ( day_ > month_days(month_) )
635  {
636  day_ -= month_days(month_);
637  if ( month_ == Month::DECEMBER )
638  year_++;
639  month_++;
640  }
641 
642  return *this;
643  }
