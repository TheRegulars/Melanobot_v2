138  bool cleanup(network::Message& msg, PropertyTree& properties ) const

155  /**
156  * \brief removes all internal nodes which don't have a name key
157  */
158  Optional<PropertyTree> restructure ( const PropertyTree& input, PropertyTree* parent ) const
159  {
160  Optional<PropertyTree> ret;
161 
162  if ( input.get_optional<std::string>("name") )
163  {
164  ret = PropertyTree();
165  parent = &*ret;

167

169  {

171  {
172  auto child = restructure(p.second,parent);
173  if ( child )

175  }
176  else if ( ret && !p.second.data().empty() )

178  ret->put(p.first,p.second.data());

204  {

206 
207  if ( opt && !opt->empty() )
208  return &*opt;

215  }

312  */
313 class Reply : public Handler
314 {
315 public:
316  Reply(const Settings& settings, handler::HandlerContainer* parent)
317  : Handler(settings,parent)
318  {
319  trigger = settings.get("trigger","");
320  reply = settings.get("reply","");
321  regex = settings.get("regex",0);
322  case_sensitive = settings.get("case_sensitive",1);
323  direct = settings.get("direct",1);
324 
325  if ( trigger.empty() || reply.empty() )
326  throw ConfigurationError();
327 
328  if ( regex )
329  {
330  auto flags = std::regex::ECMAScript|std::regex::optimize;
331  if ( !case_sensitive )
332  flags |= std::regex::icase;
333  trigger_regex = std::regex(trigger,flags);
334  }
335  }
336 
337  bool can_handle(const network::Message& msg) const override
338  {
339  return msg.type == network::Message::CHAT && (!direct || msg.direct);
340  }
341 
342 protected:
343  bool on_handle(network::Message& msg) override
344  {
345  if ( regex )
346  {
347  std::smatch match;
348  if ( std::regex_match(msg.message,match,trigger_regex) )
349  {
350  Properties map = {
351  {"sender", msg.from.name},
352  {"channel", string::implode(",",msg.channels)},
353  };
354  for ( unsigned i = 0; i < match.size(); i++ )
355  map[std::to_string(i)] = match[i];
356  on_handle(msg,string::replace(reply,map,"%"));
357  return true;
358  }
359  }
360  else if ( msg.message == trigger )
361  {
362  on_handle(msg,reply);
363  return true;
364  }
365  else if ( !case_sensitive &&
366  string::strtolower(msg.message) == string::strtolower(trigger) )
367  {
368  on_handle(msg,reply);
369  return true;
370  }
371 
372  return false;
373  }
374 
375  virtual void on_handle(const network::Message& msg, const std::string& reply) const
376  {
377  reply_to(msg,reply);
378  }
379 
380 private:
381  std::string trigger; ///< Trigger pattern
382  std::string reply; ///< Reply string
383  bool regex{false}; ///< Whether the trigger is a regex
384  bool case_sensitive{true}; ///< Whether matches are case sensitive
385  bool direct{true}; ///< Whether the input message must be direct
386  std::regex trigger_regex; ///< Regex for the trigger
