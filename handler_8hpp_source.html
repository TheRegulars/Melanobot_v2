131 class SimpleAction : public Handler
132 {
133 public:
134  /**
135  * \brief Sets up the trigger
136  * \param default_trigger Default value for \c trigger (must not be empty)
137  * \param settings Settings
138  * \param bot Pointer to the bot instance (cannot be null)
139  * \throws ConfigurationError If the requirements stated above are not met
140  */
141  SimpleAction(const std::string& default_trigger, const Settings& settings, Melanobot* bot)
142  : SimpleAction ( default_trigger, settings, bot, false ) {}
143 
144  bool can_handle(const network::Message& msg) override
145  {
146  return msg.source == source && source && authorized(msg) &&
147  ( msg.direct || !direct ) && msg.channels.size() < 2 &&
148  string::starts_with(msg.message,trigger);
149  }
150 
151  virtual bool handle(network::Message& msg)
152  {
153  if ( can_handle(msg) )
154  {
155  network::Message trimmed_msg = msg;
156  auto it = trimmed_msg.message.begin()+trigger.size();
157  it = std::find_if(it,trimmed_msg.message.end(),[](char c){return !std::isspace(c);});
158  trimmed_msg.message.erase(trimmed_msg.message.begin(),it);
159  return on_handle(trimmed_msg);
160  }
161  return false;
162  }
