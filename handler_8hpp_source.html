270  /**

272  */
273  bool can_handle(const network::Message& msg) const override

279  /**
280  * Checks with can_handle(),
281  * then creates a new message which will have the trigger stripped off
282  * and calls on_handle() with that.
283  */
284  virtual bool handle(network::Message& msg) override
285  {
286  if ( can_handle(msg) )
287  {
288  network::Message trimmed_msg = msg;
289  auto it = trimmed_msg.message.begin()+trigger.size();
290  it = std::find_if(it,trimmed_msg.message.end(),[](char c){return !std::isspace(c);});
291  trimmed_msg.message.erase(trimmed_msg.message.begin(),it);
292  return on_handle(trimmed_msg);
293  }
294  return false;
295  }

358  /**
359  * \brief Build a handler from its name and settings
360  * \return \c nullptr if it could not be created
361  */
362  std::unique_ptr<Handler> build(const std::string& handler_name,
363  const Settings& settings, handler::HandlerContainer* parent) const
364  {
365  auto it = factory.find(settings.get("type",string::strtolower(handler_name)));
366  if ( it != factory.end() )
367  {
368  try {
369  return it->second(settings, parent);
370  } catch ( const ConfigurationError& error )
371  {
372  ErrorLog("sys") << "Error creating " << handler_name << ": "
373  << error.what();
374  return nullptr;
375  }
376  }
377  ErrorLog("sys") << "Unknown handler type: " << handler_name;
378  return nullptr;
379  }
