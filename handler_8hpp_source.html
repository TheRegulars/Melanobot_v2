212 class SimpleAction : public Handler
213 {
214 public:
215  /**
216  * \brief Sets up the trigger
217  * \param default_trigger Default value for \c trigger (must not be empty)
218  * \param settings Settings
219  * \param bot Pointer to the bot instance (cannot be null)
220  * \throws ConfigurationError If the requirements stated above are not met
221  */
222  SimpleAction(const std::string& default_trigger, const Settings& settings, Melanobot* bot)
223  : SimpleAction ( default_trigger, settings, bot, false ) {}
224 
225  /**
226  * Checks authorization, \c direct, \c trigger and available reply channels.
227  */
228  bool can_handle(const network::Message& msg) const override
229  {
230  return authorized(msg) && ( msg.direct || !direct ) &&
231  msg.channels.size() < 2 && string::starts_with(msg.message,trigger);
232  }
233 
234  /**
235  * Checks with can_handle(),
236  * then creates a new message which will have the trigger stripped off
237  * and calls on_handle() with that.
238  */
239  virtual bool handle(network::Message& msg) override
240  {
241  if ( can_handle(msg) )
242  {
243  network::Message trimmed_msg = msg;
244  auto it = trimmed_msg.message.begin()+trigger.size();
245  it = std::find_if(it,trimmed_msg.message.end(),[](char c){return !std::isspace(c);});
246  trimmed_msg.message.erase(trimmed_msg.message.begin(),it);
247  return on_handle(trimmed_msg);
248  }
249  return false;
250  }
251 
252  /**
253  * Extra properties:
254  * * trigger
255  * * name (same as trigger)
256  * * direct
257  * * help
258  * * synopsis
259  */
260  std::string get_property(const std::string& name) const override
261  {
262  if ( name == "trigger" || name == "name" )
263  return trigger;
264  else if ( name == "direct" )
265  return direct ? "1" : "0";
266  else if ( name == "synopsis" )
267  return synopsis;
268  else if ( name == "help" )
269  return help;
270  return Handler::get_property(name);
271  }

273 protected:

276  bool direct = false; ///< Whether the message needs to be direct
277  /// \todo help and synopsis should be formatted strings
278  std::string synopsis; ///< Help synopsis
279  std::string help="Undocumented";///< Help string
280  bool public_reply = true;///< Whether to reply publicly or just to the sender of the message
281

284  {
285  std::string channel;
286  if ( msg.dst_channel )
287  channel = *msg.dst_channel;
288  else if ( !public_reply )
289  channel = msg.from;
290  else if ( !msg.channels.empty() )
291  channel = msg.channels[0];
292  msg.destination->say(network::OutputMessage(channel,text,priority));
293  }

432  struct RegisterHandler

435  RegisterHandler(const std::string& name, const CreateFunction& func)
436  {
437  std::string lcname = string::strtolower(name);
438  if ( HandlerFactory().instance().factory.count(lcname) )
439  ErrorLog("sys") << "Overwriting handler " << name;
440  HandlerFactory().instance().factory[lcname] = func;
441  }
442  };

450  /**
451  * \brief Build a handler from its name and settings
452  * \return \c nullptr if it could not be created
453  */
454  std::unique_ptr<Handler> build(const std::string& handler_name,
455  const Settings& settings, Melanobot* bot) const
456  {
457  auto it = factory.find(settings.get("type",string::strtolower(handler_name)));
458  if ( it != factory.end() )
459  {
460  try {
461  return it->second(settings, bot);
462  } catch ( const ConfigurationError& error )
463  {
464  ErrorLog("sys") << "Error creating " << handler_name << ": "
465  << error.what();
466  return nullptr;
467  }
468  }
469  ErrorLog("sys") << "Unknown handler type: " << handler_name;
470  return nullptr;
471  }
472 
473 private:
474  HandlerFactory() {}
475  HandlerFactory(const HandlerFactory&) = delete;
