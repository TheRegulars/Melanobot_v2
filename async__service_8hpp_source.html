132 /**
133  * \brief An AsyncService implemented with a separate thread.
134  *
135  * Handles asynchronous calls as a queue of synchronous calls performed
136  * sequentially in a separate thread.
137  *
138  * Derived classes only need to override query() and auto_load().
139  */
140 class ThreadedAsyncService : public AsyncService
141 {
142 
143  void start() override
144  {
145  requests.start();
146  if ( !thread.joinable() )
147  thread = std::move(std::thread([this]{run();}));
148  }
149 
150  void stop() override
151  {
152  requests.stop();
153  if ( thread.joinable() )
154  thread.join();
155  }
156 
157  void async_query (const Request& request, const AsyncCallback& callback) override
158  {
159  requests.push({request,callback});
160  }

162 private:
163  /**
164  * \brief Internal request record
165  */
166  struct Item
167  {
168  Request request;
169  AsyncCallback callback;
170  };
171 
172  ConcurrentQueue<Item> requests;
173  std::thread thread;
174 
175  void run()
176  {
177  while ( requests.active() )
178  {
179  Item item;
180  requests.pop(item);
181  if ( !requests.active() )
182  break;
183  item.callback(query(item.request));
184  }
