171 class AdminRaw: public SimpleAction
172 {
173 public:
174  AdminRaw(const Settings& settings, Melanobot* bot)
175  : SimpleAction("raw",settings,bot)
176  {
177  synopsis += " command";
178  help = "Sends raw IRC commands";
179  }
180 
181 protected:
182  bool on_handle(network::Message& msg) override
183  {
184  if ( !msg.message.empty() )
185  {
186  std::istringstream ss(msg.message);
187  network::Command cmd;
188  ss >> cmd.command;
189  char c;
190  std::string param;
191  while ( ss >> c )
192  {
193  if ( c == ':' )
194  {
195  std::getline(ss, param);
196  cmd.parameters.push_back(param);
197  break;
198  }
199  else
200  {
201  ss.putback(c);
202  ss >> param;
203  cmd.parameters.push_back(param);
204  }
205  }
206  msg.source->command(cmd);
207  }
208  return true;
209  }
210 };
