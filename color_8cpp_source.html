44 {

52  Component delta = cmax-cmin;
53 
54  if ( delta > 0 )

56  float hue = 0;
57  if ( r == cmax )
58  hue = float(g-b)/delta;
59  else if ( g == cmax )

61  else if ( b == cmax )
62  hue = float(r-g)/delta + 4;
63 
64  float sat = float(delta)/cmax;
65  if ( sat >= 0.3 )
66  {
67  if ( hue < 0 )
68  hue += 6;
69 
70  if ( hue <= 0.5 ) color = 0b001; // red
71  else if ( hue <= 1.5 ) color = 0b011; // yellow
72  else if ( hue <= 2.5 ) color = 0b010; // green
73  else if ( hue <= 3.5 ) color = 0b110; // cyan
74  else if ( hue <= 4.5 ) color = 0b100; // blue
75  else if ( hue <= 5.5 ) color = 0b101; // magenta
76  else color = 0b001; // red
77  }
