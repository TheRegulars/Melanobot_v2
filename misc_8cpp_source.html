130  /**
131  * \brief Remove items the user can't perform
132  * \return \b false if \c propeties shall not be considered
133  */
134  bool cleanup(network::Message& msg, PropertyTree& properties ) const
135  {
136  if ( msg.source->user_auth(msg.from,properties.get("auth","")) &&
137  properties.get("help_group",help_group) == help_group )
138  {
139  for ( auto it = properties.begin(); it != properties.end(); )
140  {
141  if ( !cleanup(msg,it->second) )
142  it = properties.erase(it);
143  else
144  ++it;
145  }
146  return true;
147  }
148  return false;
149  }

151  /**
152  * \brief removes all internal nodes which don't have a name key
153  */
154  boost::optional<PropertyTree> restructure ( const PropertyTree& input, PropertyTree* parent ) const
155  {
156  boost::optional<PropertyTree> ret;
157 
158  if ( input.get_optional<std::string>("name") )
159  {
160  ret = PropertyTree();
161  parent = &*ret;
162  }
163 
164  for ( const auto& p : input )
165  {
166  if ( !p.second.empty() )
167  {
168  auto child = restructure(p.second,parent);
169  if ( child )
170  parent->put_child(p.first,*child);
171  }
172  else if ( ret && !p.second.data().empty() )
173  {
174  ret->put(p.first,p.second.data());
175  }
176  }
177 
178  return ret;
179  }

181  /**

183  */
184  void gather(const PropertyTree& properties, std::vector<std::string>& out) const
185  {

195  /**
196  * \brief Searches for a help item with the given name
197  */
198  boost::optional<const PropertyTree&> find ( const PropertyTree& tree,
199  const std::string& name ) const
200  {
201  auto opt = tree.get_child_optional(name);
202 
203  if ( opt && !opt->empty() )
204  return opt;
205 
206  for ( const auto& p : tree )
207  {
208  opt = find(p.second,name);
209  if ( opt && !opt->empty() )
210  return opt;
211  }
212 
213  return {};
214  }

317 class Reply : public Handler
318 {
319 public:
320  Reply(const Settings& settings, Melanobot* bot)
321  : Handler(settings,bot)
322  {
323  trigger = settings.get("trigger","");
324  reply = settings.get("reply","");
325  regex = settings.get("regex",0);
326  case_sensitive = settings.get("case_sensitive",1);
327  direct = settings.get("direct",1);
328 
329  if ( trigger.empty() || reply.empty() )
330  throw ConfigurationError();
331  }
332 
333  bool can_handle(const network::Message& msg) const override
334  {
335  return Handler::can_handle(msg) && !msg.message.empty() &&
336  (!direct || msg.direct);
337  }
338 
339 protected:
340  bool on_handle(network::Message& msg) override
341  {
342  if ( regex )
343  {
344  std::smatch match;
345  auto flags = std::regex::ECMAScript;
346  if ( !case_sensitive )
347  flags |= std::regex::icase;
348  if ( std::regex_match(msg.message,match,std::regex(trigger,flags)) )
349  {
350  std::string myreply = reply;
351  if ( !match.empty() )
352  {
353  Properties map;
354  map["sender"] = msg.source->get_user(msg.from).name;
355  for ( unsigned i = 0; i < match.size(); i++ )
356  map[std::to_string(i)] = match[i];
357  myreply = string::replace(myreply,map,'\\');
358  }
359  reply_to(msg,myreply);
360  return true;
361  }
362  }
363  else if ( msg.message == trigger )
364  {
365  reply_to(msg,reply);
366  return true;
367  }
368  else if ( !case_sensitive &&
369  string::strtolower(msg.message) == string::strtolower(trigger) )
370  {
371  reply_to(msg,reply);
372  return true;
373  }
374 
375  return false;
376  }
377 
378 private:
379  std::string trigger;
380  std::string reply;
381  bool regex = false;
382  bool case_sensitive = true;
383  bool direct = true;
384 };
