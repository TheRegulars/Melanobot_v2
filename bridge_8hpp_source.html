100 /**
101  * \brief Base class for JoinMessage and similar
102  */
103 class EventMessageBase: public ::handler::Handler
104 {
105 public:
106  EventMessageBase(network::Message::Type type,
107  const Settings& settings,
108  ::handler::HandlerContainer* parent)
109  : Handler(settings,parent), type(type)
110  {
111  message = settings.get("message",message);
112  action = settings.get("action",action);
113  prefix = settings.get("prefix",prefix);
114  on_self = settings.get("on_self",on_self);
115  on_others=settings.get("on_others",on_others);
116  if ( message.empty() || !(on_others || on_self) )
117  throw ConfigurationError();
118 
119  int timeout_seconds = settings.get("timeout",0);
120  if ( timeout_seconds > 0 )
121  timeout = std::chrono::duration_cast<network::Duration>(
122  std::chrono::seconds(timeout_seconds) );
123  }
124 
125  bool can_handle(const network::Message& msg) const override
126  {
127  return msg.type == type && (
128  ( on_others && msg.from.name != msg.source->name() ) ||
129  ( on_self && msg.from.name == msg.source->name() ) );
130  }
131 
132 protected:
133  bool on_handle(network::Message& msg) override
134  {
135  string::FormatterConfig fmt;
136  auto from = msg.source->formatter()->decode(msg.from.name);
137  auto str = string::replace(message,message_replacements(msg),"%");
138  reply_to(msg,network::OutputMessage(
139  fmt.decode(str),
140  action,
141  reply_channel(msg),
142  priority,
143  action ? from : string::FormattedString(),
144  prefix,
145  timeout == network::Duration::zero() ?
146  network::Time::max() :
147  network::Clock::now() + timeout
148  ));
149  return true;
150  }
151 
152  virtual Properties message_replacements(const network::Message& msg) const
153  {
154  string::FormatterConfig fmt;
155  auto from = msg.source->formatter()->decode(msg.from.name);
156  return {
157  {"channel",string::implode(", ",msg.channels)},
158  {"name", from.encode(fmt)},
159  {"host", msg.from.host},
160  {"global_id", msg.from.global_id},
161  {"message", msg.source->formatter()->decode(msg.message).encode(fmt)}
162  };
163  }
164 
165 private:
166  network::Message::Type type;///< Type of messages to be handled
167  std::string prefix; ///< Message prefix
168  std::string message; ///< Message to send
169  bool action = false; ///< Whether it should output an action
170  bool on_self = false;///< Whether to be triggered when the joining user has the same name as the source connection
171  bool on_others=true; ///< Whether to be triggered when the joining user name differs from the source connection
172  network::Duration timeout{network::Duration::zero()};///< Output message timeout
173 };
