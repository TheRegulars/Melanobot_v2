72 protected:
73  bool on_handle(network::Message& msg) override
74  {
75  if ( msg.source->name() == msg.from )

77  msg.destination->command({"PRIVMSG",{q_bot,"users "+msg.channels[0]},priority});

79  else

81  user::User user = msg.source->get_user(msg.from);
82  if ( user.global_id.empty() )

84  }
85 
86  return false; // reacts to the message but allows to do futher processing
87  }

93 /**
94  * \brief Parses responses from Q WHOIS and USER (IRC)
95  */
96 class QGetWhois : public Handler
97 {
98 public:
99  QGetWhois(const Settings& settings, Melanobot* bot)
100  : Handler(settings,bot)
101  {
102  q_bot = settings.get("q_from",q_bot);
103  }
104 
105  bool can_handle(const network::Message& msg) const override
106  {
107  return msg.command == "NOTICE" && msg.from == q_bot &&
108  msg.source->protocol() == "irc" &&
109  msg.params.size() == 2;
110  }
111 
112 protected:
113  bool on_handle(network::Message& msg) override
114  {
115  static std::regex regex_qwhois (
116  "-Information for user (\\S+) \\(using account ([^)]+)\\):",
117  std::regex_constants::syntax_option_type::optimize |
118  std::regex_constants::syntax_option_type::ECMAScript );
119 
120  static std::regex regex_startusers (
121  "Users currently on #[^]+:",
122  std::regex_constants::syntax_option_type::optimize |
123  std::regex_constants::syntax_option_type::ECMAScript );
124 
125  static std::regex regex_users (
126  R"([ @+](\S+)\s+(\S+)\s+(?:\+[a-z]+)?\s+\([^@]+@[^)]+\))",
127  std::regex_constants::syntax_option_type::optimize |
128  std::regex_constants::syntax_option_type::ECMAScript );
129 
130  std::smatch match;
131  if ( std::regex_match(msg.params[1],match,regex_qwhois) || (
132  expects_users && std::regex_match(msg.params[1],match,regex_users) ))
133  {
134  msg.source->update_user(match[1],{{"global_id",match[2]}});
135  return true;
136  }
137  else if ( std::regex_match(msg.params[1],regex_startusers) )
138  {
139  expects_users = true;
140  return true;
141  }
142  else if ( expects_users && msg.params[1] == "End of list." )
143  {
144  expects_users = false;
145  return true;
146  }
147 
148  return false;
149  }
150 
151 private:
152  std::string q_bot = "Q";
153  bool expects_users = false; ///< Whether it's parsing the output of USERS
154 };
