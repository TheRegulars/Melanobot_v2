53 std::string build_query ( const Parameters& params )
54 {

57  for ( auto it = params.begin(); it != params.end(); ++it )

114  else if ( request.command == "PUT" )
115  response = client.put(netrequest,request.parameters);
116  else if ( request.command == "HEAD" )
