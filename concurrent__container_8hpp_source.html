28 /**
29  * \brief Container wrapper which makes it fit for concurrent use
30  *
31  * To be used with one consumer and multiple producers.
32  * It only provides utilities for insertion and extraction.
33  */
34 template <class Container>
35 class ConcurrentContainer
36 {
37 public:
38  typedef Container container_type;
39  typedef typename Container::value_type value_type;
40  typedef typename Container::reference reference;
41  typedef typename Container::const_reference const_reference;
42  typedef void(Container::*method_push)(const_reference);
43  typedef void(Container::*method_pop)();
44  typedef const_reference(Container::*method_get)() const;
45 
46  ConcurrentContainer(method_push container_push,
47  method_pop container_pop,
48  method_get container_get
49  ) : container_push(container_push), container_pop(container_pop), container_get(container_get)
50  {}
51 
52  /**
53  * \brief Add an element to the container
54  *
55  * Acquires a lock, inserts the item and notifies the consumer.
56  *
57  * If active() is not true, it will discard any input
58  */
59  void push (const_reference item)
60  {
61  if ( !run ) return;
62  std::lock_guard<std::mutex> lock(mutex);
63  (container.*container_push)(item);
64  condition.notify_one();
65  }
66 
67  /**
68  * \brief Retrieve an element from the container
69  *
70  * Waits until there are elements to get or active() is false
71  */
72  void pop(reference item)
73  {
74  std::unique_lock<std::mutex> lock(mutex);
75  condition.wait(lock,[this]{return !wait_condition();});
76  if ( !run )
77  return;
78  item = (container.*container_get)();
79  (container.*container_pop)();
80  }
81 
82  /**
83  * \brief Whether the container is allowed to process data
84  */
85  bool active() const { return run; }
86 
87  /**
88  * \brief Starts the container
89  * \note This is the state after construction
90  * \post active() is true
91  */
92  void start()
93  {
94  run = true;
95  }
96 
97  /**
98  * \brief Stops the container
99  * \post active() is false
100  */
101  void stop()
102  {
103  run = false;
104  condition.notify_one();
105  }
106 
107  /**
108  * \brief Returns a reference to the container without any locks
109  */
110  Container& get_container() { return container; }
111 
112 private:
113  method_push container_push;
114  method_pop container_pop;
115  method_get container_get;
116  Container container;
117 
118  std::atomic<bool> run {true};
119  std::mutex mutex;
120  std::condition_variable condition;
121 
122  bool wait_condition() const
123  {
124  return container.empty() && run;
125  }
126 };
