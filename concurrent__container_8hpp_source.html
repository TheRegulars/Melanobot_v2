28 /**
29  * \brief Container wrapper which makes it fit for concurrent use
30  *
31  * To be used with one consumer and multiple producers.
32  * It only provides utilities for insertion and extraction.
33  */
34 template <class Container>
35 class ConcurrentContainer
36 {
37 public:
38  using container_type = Container;

40  using reference = typename Container::reference;
41  using const_reference= typename Container::const_reference;

43  using method_pop = void(Container::*)();
44  using method_get = const_reference(Container::*)() const;

50  {}

56  /**
57  * \brief Add an element to the container
58  *
59  * Acquires a lock, inserts the item and notifies the consumer.
60  *
61  * If active() is not true, it will discard any input
62  */
63  void push (const_reference item)
64  {
65  if ( !run ) return;
66  std::lock_guard<std::mutex> lock(mutex);
67  (container.*container_push)(item);
68  condition.notify_one();
69  }
70 
71  /**
72  * \brief Retrieve an element from the container
73  *
74  * Waits until there are elements to get or active() is false
75  */
76  void pop(reference item)
77  {
78  std::unique_lock<std::mutex> lock(mutex);
79  condition.wait(lock,[this]{return !wait_condition();});
80  if ( !run )
81  return;
82  item = (container.*container_get)();
83  (container.*container_pop)();
84  }
85 
86  /**
87  * \brief Whether the container is allowed to process data
88  */
89  bool active() const { return run; }
90 
91  /**
92  * \brief Starts the container
93  * \note This is the state after construction
94  * \post active() is true
95  */
96  void start()
97  {
98  run = true;
99  }
100 
101  /**
102  * \brief Stops the container
103  * \post active() is false
104  */
105  void stop()
106  {
107  run = false;
108  condition.notify_one();
109  }
110 
111  /**
112  * \brief Returns a reference to the container without any locks
113  */
114  Container& get_container() { return container; }
115 
116 private:
117  method_push container_push;
118  method_pop container_pop;
119  method_get container_get;
120  Container container;
121 
122  std::atomic<bool> run {true};
123  std::mutex mutex;
124  std::condition_variable condition;
125 
126  bool wait_condition() const
127  {
128  return container.empty() && run;
129  }
