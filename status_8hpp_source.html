446  /**

448  */
449  void handle_end(const network::Message& msg, const std::smatch& match)

451  string::FormatterConfig fmt;

464  /**
465  * \brief Prints the score for all the players in a given team
466  */
467  void print_scores(const network::Message& msg, int team)
468  {
469  auto it = player_scores.find(team);
470  if ( it == player_scores.end() || it->second.empty() )
471  return;
472 
473  if ( team != SPECTATORS )
474  {
475  if ( sort_reverse )
476  std::stable_sort(it->second.begin(),it->second.end());
477  else
478  std::stable_sort(it->second.rbegin(),it->second.rend());
479  }
480 
481  string::FormattedString out;
482  auto color = team_colors[team];
483 
484  auto itt = team_scores.find(team);
485  if ( itt != team_scores.end() )
486  out << color << itt->second << color::nocolor << ") ";
487 
488  for ( unsigned i = 0; i < it->second.size(); i++ )
489  {
490  if ( team != SPECTATORS )
491  out << color << string::FormatFlags::BOLD << it->second[i].score
492  << string::ClearFormatting() << ' ';
493 
494  out << msg.source->formatter()->decode(it->second[i].name)
495  << string::ClearFormatting();
496 
497  if ( i < it->second.size() - 1 )
498  out << ", ";
499  }
500 
501  reply_to(msg, out);
502  }
