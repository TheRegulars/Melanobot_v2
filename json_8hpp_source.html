245  /**
246  * \brief Parse a string literal
247  */
248  std::string parse_string()
249  {
250  char c = get_skipws();
251  std::string r;
252  if ( c == '\"' )
253  {
254  while ( true )
255  {
256  c = stream.get();
257  if ( !stream || c == '\"' )
258  break;
259  if ( c == '\\' )
260  {
261  c = escape(stream.get());
262  if ( !stream )
263  break;
264 
265  if ( c == 'u' )
266  {
267  char hex[] = "0000";
268  stream.read(hex,4);
269  for ( int i = 0; i < 4; i++ )
270  {
271  if ( !std::isxdigit(hex[i]) )
272  hex[i] = '0';
273  }
274  r += string::Utf8Parser::encode(std::stoul(hex,0,16));
275 
276  continue;
277  }
278 
279  if ( !escapeable(c) )
280  r += '\\';
281  }
282 
283  r += c;
284 
285  if ( c == '\n' )
286  line++;
287  }
288  }
289  else
290  {
291  while ( true )
292  {
293  r += c;
294  c = stream.get();
295  if ( !stream || ( !std::isalnum(c) && c != '_' && c != '-' ) )
296  break;
297  }
298  stream.unget();
299  }
300 
301  return r;
