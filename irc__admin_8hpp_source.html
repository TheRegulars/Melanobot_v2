148  bool on_handle(network::Message& msg) override
149  {
150  if ( !msg.message.empty() )
151  {
152  std::istringstream ss(msg.message);
153  network::Command cmd;
154  ss >> cmd.command;
155  char c;
156  std::string param;
157  while ( ss >> c )
158  {
159  if ( c == ':' )
160  {
161  std::getline(ss, param);
162  cmd.parameters.push_back(param);
163  break;
164  }
165  else
166  {
167  ss.putback(c);
168  ss >> param;
169  cmd.parameters.push_back(param);
170  }
171  }
172  msg.destination->command(cmd);
173  }
174  return true;
175  }
