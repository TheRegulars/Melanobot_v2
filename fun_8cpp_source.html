346 protected:
347  bool on_handle(network::Message& msg) override
348  {
349  using namespace boost::filesystem;
350 
351  if ( exists(pony_path) && is_directory(pony_path))
352  {
353  std::string pony_search = msg.message;
354  int score = -1;
355  std::vector<std::string> files;
356 
357  for( directory_iterator dir_iter(pony_path) ;
358  dir_iter != directory_iterator() ; ++dir_iter)
359  {
360  if ( is_regular_file(dir_iter->status()) )
361  {
362  if ( pony_search.empty() )
363  {
364  files.push_back(dir_iter->path().string());
365  }
366  else
367  {
368  int local_score = string::similarity(
369  dir_iter->path().filename().string(), pony_search );
370  if ( local_score > score )
371  {
372  score = local_score;
373  files = { dir_iter->path().string() };
374  }
375  else if ( local_score > score )
376  {
377  files.push_back( dir_iter->path().string() );
378  }
379  }
380  }
381  }
382 
383  if ( !files.empty() )
384  {
385  std::fstream file(files[math::random(files.size()-1)]);
386  if ( file.is_open() )
387  {
388  std::string line;
389  while(true)
390  {
391  std::getline(file,line);
392  if ( !file ) break;
393  reply_to(msg,line);
394  }
395  return true;
396  }
397  }
398  }
399 
400  reply_to(msg,"Didn't find anypony D:");
401  return true;
402  }
