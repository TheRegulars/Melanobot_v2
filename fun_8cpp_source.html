352  bool on_handle(network::Message& msg) override
353  {
354  using namespace boost::filesystem;
355 
356  if ( exists(pony_path) && is_directory(pony_path))
357  {
358  std::string pony_search = msg.message;
359  int score = -1;
360  std::vector<std::string> files;
361 
362  // For all of the files in the given directory
363  for( directory_iterator dir_iter(pony_path) ;
364  dir_iter != directory_iterator() ; ++dir_iter)
365  {
366  // which are regular files
367  if ( is_regular_file(dir_iter->status()) )
368  {
369  if ( pony_search.empty() )
370  {
371  // no search query? any pony will do
372  files.push_back(dir_iter->path().string());
373  }
374  else
375  {
376  // get how similar the query is to the file name
377  int local_score = string::similarity(
378  dir_iter->path().filename().string(), pony_search );
379 
380  if ( local_score > score )
381  {
382  // found a better match, use that
383  score = local_score;
384  files = { dir_iter->path().string() };
385  }
386  else if ( local_score == score )
387  {
388  // found an equivalent match, add it to the list
389  files.push_back( dir_iter->path().string() );
390  }
391  }
392  }
393  }
394 
395  // found at least one pony
396  if ( !files.empty() )
397  {
398  // open a random one
399  std::fstream file(files[math::random(files.size()-1)]);
400  // I guess if not is_open and we have other possible ponies
401  // we could select a different one (random_shuffle and iterate)
402  // but whatever
403  if ( file.is_open() )
404  {
405  // print the file line by line
406  std::string line;
407  while(true)
408  {
409  std::getline(file,line);
410  if ( !file ) break;
411  reply_to(msg,line);
412  }
413  // done, return
414  return true;
415  }
416  }
417  }
418 
419  // didn't find any suitable file
420  reply_to(msg,"Didn't find anypony D:");
421  return true;
422  }
