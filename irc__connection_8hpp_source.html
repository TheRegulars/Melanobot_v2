43 struct LoginInfo

46  std::string password; ///< Password used to log in
47  std::string service; ///< Service used to log in
48  std::string command; ///< Command used to log in

51  /**
52  * \brief Read from settings
53  */
54  explicit LoginInfo(const Settings& settings, const std::string& nick="");
55 
56  /**
57  * \brief Whether it has all the required information to auth
58  */
59  bool can_auth() const
60  {
61  return !nick.empty() && !password.empty() &&
62  !service.empty() && !command.empty();
63  }
64 
65  /**
66  * \brief Builds the auth command
67  *
68  * PRIVMSG \c service :\c command \c nick \c password
69  */
70  Command irc_command(int priority) const
71  {
72  return {"PRIVMSG",{service,command+' '+nick+' '+password},priority};
73  }
74 };
