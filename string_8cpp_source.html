69 void Utf8Parser::parse(const std::string& string)
70 {
71  input.str(string);
72  while ( !input.eof() )
73  {
74  uint8_t byte = input.next();
75 
76  // 0... .... => ASCII
77  if ( byte < 0b1000'0000 )
78  {
79  check_valid();
80  callback(callback_ascii,byte);
81  }
82  // 11.. .... => Begin multibyte
83  else if ( (byte & 0b1100'0000) == 0b1100'0000 )
84  {
85  check_valid();
86  utf8.push_back(byte);
87 
88  // extract number of leading 1s
89  while ( byte & 0b1000'0000 )
90  {
91  length++;
92  byte <<= 1;
93  }
94 
95  // Restore byte (leading 1s have been eaten off)
96  byte >>= length;
97  unicode = byte;
98  }
99  // 10.. .... => multibyte tail
100  else if ( length > 0 )
101  {
102  utf8.push_back(byte);
103  unicode <<= 6;
104  unicode |= byte&0b0011'1111; //'
105  if ( utf8.size() == length )
106  {
107  callback(callback_utf8,unicode,utf8);
108  unicode = 0;
109  length = 0;
110  utf8.clear();
111  }
112  }
113  }
114  check_valid();
115  callback(callback_end);
116 }

118 std::string Utf8Parser::encode(uint32_t value)
119 {
120  if ( value < 128 )
121  return std::string(1,char(value));
122 
123  std::list<uint8_t> s;
124 
125  uint8_t head = 0;
126  while ( value )
127  {
128  s.push_back((value&0b0011'1111)|0b1000'0000);
129  value >>= 6;
130  head <<= 1;
131  head |= 1;
132  }
133 
134  if ( (uint8_t(s.back())&0b0011'1111) > (1 << (7 - s.size())) )
135  {
136  head <<= 1;
137  head |= 1;
138  s.push_back(0);
139  }
140 
141  s.back() |= head << (8 - s.size());
142 
143  return std::string(s.rbegin(),s.rend());
144 }

327 FormattedString FormatterAnsi::decode(const std::string& source) const
328 {
329  FormattedString str;
330 
331  Utf8Parser parser;
332 
333  std::string ascii;
334 
335  auto push_ascii = [&ascii,&str]()
336  {
337  if ( !ascii.empty() )
338  {
339  str.append<AsciiSubstring>(ascii);
340  ascii.clear();
341  }
342  };
343 
344  parser.callback_ascii = [&parser,&str,&ascii,push_ascii](uint8_t byte)
345  {
346  if ( byte == '\x1b' && parser.input.peek() == '[' )
347  {
348  push_ascii();
349  parser.input.ignore();
350  std::vector<int> codes;
351  while (parser.input)
352  {
353  int i = 0;
354  if ( parser.input.get_int(i) )
355  {
356  codes.push_back(i);
357  char next = parser.input.next();
358  if ( next != ';' )
359  {
360  if ( next != 'm' )
361  parser.input.unget();
362  break;
363  }
364  }
365  }
366  FormatFlags flags;
367  bool use_flags = false;
368  for ( int i : codes )
369  {
370  if ( i == 0)
371  {
372  str.append<ClearFormatting>();
373  }
374  else if ( i == 1 )
375  {
376  flags |= FormatFlags::BOLD;
377  use_flags = true;
378  }
379  else if ( i == 22 )
380  {
381  flags &= ~FormatFlags::BOLD;
382  use_flags = true;
383  }
384  else if ( i == 4 )
385  {
386  flags |= FormatFlags::UNDERLINE;
387  use_flags = true;
388  }
389  else if ( i == 24 )
390  {
391  flags &= ~FormatFlags::UNDERLINE;
392  use_flags = true;
393  }
394  else if ( i == 39 )
395  {
396  str.append<Color>(color::nocolor);
397  }
398  else if ( i >= 30 && i <= 37 )
399  {
400  i -= 30;
401  if ( flags == FormatFlags::BOLD )
402  {
403  i |= 0b1000;
404  flags = FormatFlags::NO_FORMAT;
405  use_flags = false;
406  }
407  str.append<Color>(color::Color12::from_4bit(i));
408  }
409  else if ( i >= 90 && i <= 97 )
410  {
411  i -= 90;
412  i |= 0b1000;
413  str.append<Color>(color::Color12::from_4bit(i));
414  }
415  }
416  if ( use_flags )
417  str.append<Format>(flags);
418  }
419  else
420  {
421  ascii += byte;
422  }
423  };
424  parser.callback_utf8 = [this,&str,push_ascii](uint32_t unicode,const std::string& utf8)
425  {
426  push_ascii();
427  if ( this->utf8 )
428  str.append<Unicode>(utf8,unicode);
429  };
430  parser.callback_end = push_ascii;
431 
432  parser.parse(source);
433 
434  return str;
435 }
