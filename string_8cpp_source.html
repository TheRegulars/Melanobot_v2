70 void Utf8Parser::parse(const std::string& string)
71 {
72  input.clear();
73  input.str(string);
74  while ( true )
75  {
76  uint8_t byte = input.get();
77 
78  if ( !input )
79  break;
80 
81  // 0... .... => ASCII
82  if ( byte < 0b1000'0000 )
83  {
84  check_valid();
85  callback(callback_ascii,byte);
86  }
87  // 11.. .... => Begin multibyte
88  else if ( (byte & 0b1100'0000) == 0b1100'0000 )
89  {
90  check_valid();
91  utf8.push_back(byte);
92 
93  // extract number of leading 1s
94  while ( byte & 0b1000'0000 )
95  {
96  length++;
97  byte <<= 1;
98  }
99 
100  // Restore byte (leading 1s have been eaten off)
101  byte >>= length;
102  unicode = byte;
103  }
104  // 10.. .... => multibyte tail
105  else if ( length > 0 )
106  {
107  utf8.push_back(byte);
108  unicode <<= 6;
109  unicode |= byte&0b0011'1111; //'
110  if ( utf8.size() == length )
111  {
112  callback(callback_utf8,unicode,utf8);
113  unicode = 0;
114  length = 0;
115  utf8.clear();
116  }
117  }
118  }
119  check_valid();
120  callback(callback_end);
121 }

123 std::string Utf8Parser::encode(uint32_t value)
124 {
125  if ( value < 128 )
126  return std::string(1,char(value));
127 
128  std::list<uint8_t> s;
129 
130  uint8_t head = 0;
131  while ( value )
132  {
133  s.push_back((value&0b0011'1111)|0b1000'0000);
134  value >>= 6;
135  head <<= 1;
136  head |= 1;
137  }
138 
139  if ( (uint8_t(s.back())&0b0011'1111) > (1 << (7 - s.size())) )
140  {
141  head <<= 1;
142  head |= 1;
143  s.push_back(0);
144  }
145 
146  s.back() |= head << (8 - s.size());
147 
148  return std::string(s.rbegin(),s.rend());
149 }

335 FormattedString FormatterAnsi::decode(const std::string& source) const
336 {
337  FormattedString str;
338 
339  Utf8Parser parser;
340 
341  std::string ascii;
342 
343  auto push_ascii = [&ascii,&str]()
344  {
345  if ( !ascii.empty() )
346  {
347  str.append<AsciiSubstring>(ascii);
348  ascii.clear();
349  }
350  };
351 
352  parser.callback_ascii = [&parser,&str,&ascii](uint8_t byte)
353  {
354  if ( byte == '\x1b' && parser.input.peek() == '[' )
355  {
356  parser.input.ignore();
357  std::vector<int> codes;
358  while (parser.input)
359  {
360  int i = 0;
361  if ( parser.input >> i )
362  {
363  codes.push_back(i);
364  char next = parser.input.get();
365  if ( next != ';' )
366  {
367  if ( next != 'm' )
368  parser.input.unget();
369  break;
370  }
371  }
372  }
373  FormatFlags flags;
374  bool use_flags = false;
375  for ( int i : codes )
376  {
377  if ( i == 0)
378  {
379  str.append<ClearFormatting>();
380  }
381  else if ( i == 1 )
382  {
383  flags |= FormatFlags::BOLD;
384  use_flags = true;
385  }
386  else if ( i == 22 )
387  {
388  flags &= ~FormatFlags::BOLD;
389  use_flags = true;
390  }
391  else if ( i == 4 )
392  {
393  flags |= FormatFlags::UNDERLINE;
394  use_flags = true;
395  }
396  else if ( i == 24 )
397  {
398  flags &= ~FormatFlags::UNDERLINE;
399  use_flags = true;
400  }
401  else if ( i == 39 )
402  {
403  str.append<Color>(color::nocolor);
404  }
405  else if ( i >= 30 && i <= 37 )
406  {
407  i -= 30;
408  if ( flags == FormatFlags::BOLD )
409  {
410  i &= 0xf;
411  flags = FormatFlags::NO_FORMAT;
412  use_flags = false;
413  }
414  str.append<Color>(color::Color12::from_4bit(i));
415  }
416  else if ( i >= 90 && i <= 97 )
417  {
418  i -= 90;
419  i &= 0xf;
420  str.append<Color>(color::Color12::from_4bit(i));
421  }
422  }
423  if ( use_flags )
424  str.append<Format>(flags);
425  }
426  else
427  {
428  ascii += byte;
429  }
430  };
431  parser.callback_utf8 = [this,&str,push_ascii](uint32_t unicode,const std::string& utf8)
432  {
433  push_ascii();
434  if ( this->utf8 )
435  str.append<Unicode>(utf8,unicode);
436  };
437  parser.callback_end = push_ascii;
438 
439  parser.parse(source);
440 
441  return str;
442 }
