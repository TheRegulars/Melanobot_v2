42  /**
43  * \brief Parses a time point
44  * \code
45  * TIME_POINT ::= NOW_TIME | DATE_TIME
46  * NOW_TIME ::= now
47  * | now + DURATION
48  * | now - DURATION
49  * \endcode
50  */
51  DateTime parse_time_point()
52  {
53  if ( lookahead.type == Token::NOW )
54  {
55  DateTime now;
56  scan();
57  if ( lookahead.type == Token::OPERATOR )
58  {
59  char oper = lookahead.lexeme[0];
60  scan();
61  if ( oper == '+' )
62  now += parse_duration();
63  else if ( oper == '-' )
64  now -= parse_duration();
65  }
66  return now;
67  }
68  return parse_date_time();
69  }

71  /**
72  * \brief Parses a duration
73  * \code
74  * DURATION ::= HOUR_OPT DURATION_SEQ
75  * HOUR_OPT ::= (eps) | time | time "h" | time "min"
76  * DURATION_SEQ ::= ATOMIC_DURATION | ATOMIC_DURATION DURATION_SEQ | (eps)
77  * \endcode
78  */
79  Duration parse_duration()
80  {
81  Duration duration = Duration::zero();
82 
83  // check of the optional time
84  if ( lookahead.type == Token::TIME )
85  {
86  // get the time
87  duration = token_val<Duration>();
88  // 12:34 is a valid minute, 12:34:56 is not
89  bool can_be_minutes = lookahead.lexeme.size() == 5;
90  // advance lookahead
91  scan();
92  // check for the optional unit (hours or minutes)
93  if ( lookahead.type == Token::IDENTIFIER )
94  {
95  if ( is_unit<hours>(lookahead.lexeme) )
96  {
97  // hours, nothing to do
98  scan();
99  }
100  else if ( can_be_minutes && is_unit<minutes>(lookahead.lexeme) )
101  {
102  // hours become minutes
103  duration /= 60;
104  scan();
105  }
106  }
107  }
108 
109  // Read multiple number/unit pairs
110  while ( lookahead.type == Token::NUMBER )
111  duration += parse_atomic_duration();
112 
113  return duration;
114  }

156  * \param c Lookahead character
157  */

159  {
160  std::string id;

162  {

169

174  if ( lower == "tomorrow" )
175  return {Token::REL_DAY, id, 1};
176  if ( lower == "yesterday" )
177  return {Token::REL_DAY, id, -1};
178  if ( lower == "am" )
179  return {Token::AMPM, id, false};
180  if ( lower == "pm" )
181  return {Token::AMPM, id, true};

183  auto maybe_month = month_from_name(lower);
184  if ( maybe_month )

186 
187  auto maybe_weekday = weekday_from_name(lower);
188  if ( maybe_weekday )
189  return {Token::WEEK_DAY, id, *maybe_weekday};
190 
191  return {Token::IDENTIFIER, id, std::move(lower)};

194  /**
195  * \brief Lexes a time (12:34[:56[.789]])
196  * \param lexed Lexed first two digits
197  * \pre \c lexed contains two decimal digits
198  * \pre A ':' has been lexed
199  */
200  Token lex_time(std::string& lexed)
201  {

203  dur += hours(string::to_uint(lexed));

205  char c = input.get();
206  std::string part2_lex = lex_raw_number(c);
207  dur += minutes(string::to_uint(part2_lex));

213  c = input.get();

215  dur += seconds(string::to_uint(part2_lex));

293  /**
294  * \brief Reads the next token
295  */
296  Token lex()
297  {
298  char c = ' ';
299  while ( (std::isspace(c) || c == ',') && input )
300  c = input.get();
301  if ( !input )
302  return {};
303 
304  if ( std::isalpha(c) )
305  return lex_identifier(c);
306  else if ( std::isdigit(c) )
307  return lex_number(c);
308  else if ( c == '+' || c == '-' )
309  return {Token::OPERATOR, std::string(1,c)};
310  else if ( c == '\'' || c == '"' ) // minute/second unit
311  return {Token::IDENTIFIER, std::string(1,c), std::string(1,c)};
312 
313  return {};
314  }

342  * \tparam DurationT one of the type aliases like \c seconds and friends
343  */
344  template <class DurationT>
345  static bool is_unit(const std::string& str)
346  {
347  if ( std::is_same<DurationT,days>::value )
348  return string::is_one_of(str,{"day","days","d"});
349  if ( std::is_same<DurationT,hours>::value )
350  return string::is_one_of(str,{"hours","hour","h"});
351  if ( std::is_same<DurationT,minutes>::value )
352  return string::is_one_of(str,{"minutes","minute","m","min","'"});
353  if ( std::is_same<DurationT,seconds>::value )
354  return string::is_one_of(str,{"seconds","second","s","\""});
355  if ( std::is_same<DurationT,milliseconds>::value )
356  return string::is_one_of(str,{"milliseconds","millisecond","ms"});
357  return false;
358  }

361  * \brief Parses an atomic duration
362  * \code
363  * ATOMIC_DURATION ::= number unit
364  * \endcode
365  */
366  Duration parse_atomic_duration()
367  {
368  // check lookahead is a number
369  if ( lookahead.type == Token::NUMBER )
370  {
371  // get the value
372  Duration::rep ticks = token_val<uint32_t>();
373  // advance lookahead to the next token
374  scan();
375  // now we should have a unit identifier
376  if ( lookahead.type == Token::IDENTIFIER )
377  {
378  // check the known units
379  if ( is_unit<milliseconds>(token_val<std::string>()) )
380  {
381  scan();
382  return duration_cast(milliseconds(ticks));
383  }
384  else if ( is_unit<seconds>(token_val<std::string>()) )
385  {
386  scan();
387  return duration_cast(seconds(ticks));
388  }
389  else if ( is_unit<minutes>(token_val<std::string>()) )
390  {
391  scan();
392  return duration_cast(minutes(ticks));
393  }
394  else if ( is_unit<hours>(token_val<std::string>()) )
395  {
396  scan();
397  return duration_cast(hours(ticks));
398  }
399  else if ( is_unit<days>(token_val<std::string>()) )
400  {
401  scan();
402  return duration_cast(days(ticks));
403  }
404  }
405  }
406  // fail

454  * DATE_DESC ::= OPT_WEEK_DAY month MONTH_DAY OPT_YEAR

551  /**
552  * \brief Parses a time and writes it to \c out
553  * \code
554  * TIME ::= time | time AMPM
555  * \endcode
556  */
557  void parse_time(DateTime& out)
558  {
559  if ( lookahead.type != Token::TIME )
560  return;
561  Duration time = token_val<Duration>();
562 
563  hours hour = std::chrono::duration_cast<hours>(time);
564  if ( hour.count() > 24 ) return; // invalid hour
565  time -= hour;
566 
567  minutes mins = std::chrono::duration_cast<minutes>(time);
568  if ( mins.count() >= 60 ) return; // invalid minute
569  time -= mins;
570 
571  seconds second = std::chrono::duration_cast<seconds>(time);
572  time -= second;
573 
574  milliseconds millisecond = std::chrono::duration_cast<milliseconds>(time);
575 
576  scan();

578  {

582  hour += hours(12);
